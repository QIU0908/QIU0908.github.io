LPF_GY85_2_servo(close_loop)

#include <Wire.h>
#include <Servo.h>

Servo myServo;

unsigned long lastTime = 0;

#define ADXL345 0x53

int16_t offset_x = 0, offset_y = 0, offset_z = 0;

// ================== 濾波相關變數 ==================
float filtered_roll = 0.0;        // 濾波後的 roll
float alpha_roll = 0.25;          // 指令端 alpha（可調）

// ================== Servo ADC 濾波 ==================
const int median_size = 9;
int adc_buffer[median_size] = {0};//等於0矩陣
int buffer_idx = 0;
float filtered_adc_angle = 90.0;
float alpha_adc = 0.18;           // ADC 端 alpha 為了讓實際角度讀值平滑，去除馬達負載造成的凸波（spike）和雜訊（主要用在分析）
//越小  濾波越強，越大 減少延遲

// ================== PID 控制變數 ==================
float Kp = 2.;     //  P 值
float Ki = 0.003;    // 積分增益
float Kd = 0.0;     // 微分增益 但 alpha_adc 已有率波功能所以這裡先不用

float integral = 0.0;         // 積分項累積
float last_error = 0.0;       // 上一次誤差（用於微分）
float error_deadband = 5.;   // 死區 之前 角度 to ADC 誤差差不多5度

float prev_target = 0;        // 偵測目標角度有沒有變化

// =================================================
int16_t readAxis(byte reg);
void cailbrate();

void setup() {
  Serial.begin(115200);
  Wire.setSCL(PB8);
  Wire.setSDA(PB9);
  Wire.begin();
  delay(1000);

  // ================== 初始化 ADXL345 ================== 
  Wire.beginTransmission(ADXL345);
  Wire.write(0x2D); Wire.write(0x08); Wire.endTransmission();
  Wire.beginTransmission(ADXL345);
  Wire.write(0x31); Wire.write(0x00); Wire.endTransmission();

  Serial.println("ADXL345 ready");

  myServo.attach(PA0);
  myServo.write(90);
  Serial.println("Servo ready");

  cailbrate();

  delay(500);
  lastTime = micros();
}
void loop() {
  if(micros() - lastTime >= 20000){   // 20ms 一次 SG90 更新 ≈ 50Hz
    lastTime += 20000;

    // ================== 讀取 ADXL345 ==================
    float x = readAxis(0x32) - offset_x;
    float y = readAxis(0x34) - offset_y;
    float z = readAxis(0x36) - offset_z;

    float x_g = x / 256.0;
    float y_g = y / 256.0;
    float z_g = z / 256.0;

    float raw_roll = atan2(y_g, sqrt(x_g * x_g + z_g * z_g)) * 180.0 / PI;

    // ==================濾波==================
    filtered_roll = alpha_roll * raw_roll + (1 - alpha_roll) * filtered_roll;

    // ==================開迴路指令角度==================
    filtered_roll = constrain(filtered_roll, -60, 60);//filtered_roll範圍±60 

    float GY_TO_servo_angle =
        (filtered_roll + 60.0) * (150.0 - 30.0) / 120.0 + 30.0;//把filtered_roll範圍改成0~120
        //線性映射公式y = (x - xmin) * (ymax - ymin) / (xmax - xmin) + ymin
        //(x - xmin)/ (xmax - xmin)把 x 壓到0~1
        //× (ymax - ymin)放大到新範圍
        //+ ymin +平移
        GY_TO_servo_angle = constrain(GY_TO_servo_angle, 30, 150);

    // ================== ADC 讀取與濾波 ==================
    int raw_adc = analogRead(PA7);
    adc_buffer[buffer_idx] = raw_adc;
    buffer_idx = (buffer_idx + 1) % median_size;

    int temp[median_size];
    for (int i = 0; i < median_size; i++) temp[i] = adc_buffer[i];
    sort(temp, median_size);
    int median_adc = temp[median_size / 2];
    
    float adc_angle = 0.215583 * median_adc - 18.501444 -8.; //加上offset 偏移

    filtered_adc_angle = alpha_adc * adc_angle + (1 - alpha_adc) * filtered_adc_angle;

    // ================== 誤差計算 ==================

    if (abs(GY_TO_servo_angle - prev_target) > 2.0) {
      integral = 0;  // 目標變了，清掉積分
      }
      prev_target = GY_TO_servo_angle;

    float error = GY_TO_servo_angle - filtered_adc_angle;     // 誤差 = 指令角度 - 濾波後實際角度

    if(abs(error) < error_deadband){ //防小誤差一直積分
      error = 0;
      integral *= 0.85;  // 死區內積分衰減
      }
    else{
      error -= (error > 0 ? error_deadband : -error_deadband);
      }

    // ================== PID 計算 ==================
    integral += error * 0.02;                    // 0.02 秒 = 20ms
    float derivative = (error - last_error) / 0.02;

    // 限制積分項（抗積分飽和）
    integral = constrain(integral, -15, 15);

    float pid_output = Kp * error + Ki * integral + Kd * derivative;

   // ================== 最終輸出 ==================
    int compensated_angle = (int)(GY_TO_servo_angle + pid_output);
    compensated_angle = constrain(compensated_angle, 25, 155);                       
    

    // ==================驅動 Servo==================
    myServo.write(compensated_angle);
   

    // ================== 序列埠輸出（方便調參） ==================
    Serial.print(GY_TO_servo_angle, 1);           // 指令角度
    Serial.print(",");
    Serial.print(filtered_adc_angle, 1);          // 實際角度
    Serial.print(",");
    Serial.print(error, 2);                       // 誤差
    Serial.print(",");
    Serial.print(pid_output, 2);                  // PID 輸出值
    Serial.print(",");
    Serial.println(compensated_angle);            // 最終送給 servo 的角度

    last_error = error;
  }
}

// ================== 輔助函式 ==================
int16_t readAxis(byte reg) {
  Wire.beginTransmission(ADXL345); 
  Wire.write(reg);  // 指向 DATAX0 暫存器（X 軸低位元組）
  Wire.endTransmission(false);// 保持連線，準備讀取
  Wire.requestFrom(ADXL345, 2);
    // 請求 2 個位元組（LSB + MSB）
    // I2C 每次只能傳 8 位元，所以 16 位元的 X 軸數值
    // 被 ADXL345 拆成兩個暫存器分兩次傳送
  if (Wire.available() >= 2) {// 確認 2 個位元組都已收到
    byte lsb = Wire.read();// 第一次讀 → DATAX0 低位元組
    byte msb = Wire.read();// 第二次讀 → DATAX1 高位元組
    // 將兩個位元組重新拼合成完整的 16 位元數值
    // 為什麼要拆開又合起來？
    // 因為 I2C 硬體限制每次只能傳 8 位元
    // ADXL345 被迫把 16 位元數值拆成兩半分兩次送出
    // 所以我們收到後必須自己拼回來：
    //
    return (int16_t)((msb << 8) | lsb);
  }
  return 0;
}

//校正階段 找offset
void cailbrate() {
  Serial.print("校正中 保持靜止...");
  long sum_x = 0, sum_y = 0, sum_z = 0;
  for (int i = 0; i < 500; i++) {
    sum_x += readAxis(0x32);
    sum_y += readAxis(0x34);
    sum_z += readAxis(0x36);
    delay(10);
  }
  offset_x = sum_x / 500;
  offset_y = sum_y / 500;
  offset_z = sum_z / 500 - 256;
  Serial.println("校正完成");
}

// 小排序函式（用於中位數濾波）
void sort(int arr[], int n) {
  for (int i = 0; i < n - 1; i++) {
    for (int j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        int temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
  }
}
