#include<Wire.h>
#include<Servo.h>

Servo myServo;

unsigned long lastTime = 0;


#define ADXL345 0x53

int16_t offset_x =0;
int16_t offset_y =0;
int16_t offset_z =0;



int16_t readAxis(byte );
void cailbrate();

void setup(){
  Serial.begin(115200);

  Wire.setSCL(PB8);
  Wire.setSDA(PB9);
  Wire.begin();
  delay(1000);

  Wire.beginTransmission(ADXL345);
  Wire.write(0x2D);
  Wire.write(0x08);
  Wire.endTransmission();

  Wire.beginTransmission(ADXL345);
  Wire.write(0x31);
  Wire.write(0x00);
  Wire.endTransmission();

  Serial.println("ADXL345 ready");

  delay(100);
  
  myServo.attach(PA0);
  myServo.write(90);
  Serial.println("Servo ready");
  delay(1000);

  cailbrate();
  Serial.println("millis  GY_TO_servo_angle   ADC  ");
  delay(100);
}


void loop() {
   if(micros() - lastTime >= 10000){ // 10000us = 10ms
  lastTime += 10000;
    
    // 讀取三軸並減去偏移
    float x = readAxis(0x32) - offset_x;
    float y = readAxis(0x34) - offset_y;
    float z = readAxis(0x36) - offset_z;

    // 換算成 g（±2g 量程，256 LSB/g）
    float x_g = x / 256.0;
    float y_g = y / 256.0;
    float z_g = z / 256.0;

    // 計算 Pitch 和 Roll
    // Pitch：前後傾斜角度
    // Roll ：左右傾斜角度
    float pitch = atan2(x_g, sqrt(y_g * y_g + z_g * z_g)) * 180.0 / PI;
    float roll  = atan2(y_g, sqrt(x_g * x_g + z_g * z_g)) * 180.0 / PI;

    int GY_TO_servo_angle = map(constrain((int)roll,-90,90),-90,90 ,30,150);
// constrain (輸入值, 最小, 最大)     EX:輸入值<最小 ,輸入值=最小
// map (輸入值, 輸入最小值, 輸入最大值, 輸入最小值對應到的輸出最小值, 輸入最大值對應到的輸出最大值)
myServo.write(GY_TO_servo_angle);

Serial.print(millis());
Serial.print(",");
Serial.print(GY_TO_servo_angle);
Serial.print(",");
Serial.println(analogRead(PA7));

    }
}

int16_t readAxis(byte reg){
  Wire.beginTransmission(ADXL345);
  Wire.write(reg);
  Wire.endTransmission(false);
  Wire.requestFrom(ADXL345,2);
  if(Wire.available()>=2){
    byte lsb =Wire.read();
    byte msb =Wire.read();
    return (int16_t)((msb << 8) | lsb); 
  }else
    return 0;
}

void cailbrate(){
  Serial.print("校正中 保持靜止");
  long sum_x=0,sum_y=0,sum_z=0;

  for(int i=0;i<500;i++){
    sum_x += readAxis(0x32);
    sum_y += readAxis(0x34);
    sum_z += readAxis(0x36);
    delay(10);
  }
  offset_x =sum_x /500;
  offset_y =sum_y /500;
  offset_z =sum_z /500 - 256;//水平時Z為256(1g)

  Serial.println("校正完成");
}
