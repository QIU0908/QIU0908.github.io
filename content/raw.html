#include <Wire.h>
#include <Servo.h>

Servo myServo;

unsigned long lastTime = 0;

#define ADXL345 0x53

int16_t offset_x = 0, offset_y = 0, offset_z = 0;

// ================== жҝҫжіўзӣёй—ңи®Ҡж•ё ==================
float filtered_roll = 0.0;
float alpha_roll = 0.35;   //alpha и¶Ҡе°Ҹ жҝҫжіўи¶Ҡеј· е»¶йҒІи¶ҠеӨ§ еҸҚд№Ӣ

void setup() {
  Serial.begin(115200);
  Wire.setSCL(PB8);
  Wire.setSDA(PB9);
  Wire.begin();
  delay(1000);

  //==================еҲқе§ӢеҢ– ADXL345==================
  Wire.beginTransmission(ADXL345);
  Wire.write(0x2D); Wire.write(0x08); Wire.endTransmission();
  Wire.beginTransmission(ADXL345);
  Wire.write(0x31); Wire.write(0x00); Wire.endTransmission();

  Serial.println("ADXL345 ready");

  myServo.attach(PA0);
  myServo.write(90);
  Serial.println("Servo ready");

  cailbrate();   
  delay(500);
  lastTime = micros();
}

void loop() {
  if (micros() - lastTime >= 20000) {   // 50Hz жӣҙж–°
    lastTime += 20000;

    //==================и®ҖеҸ– ADXL345==================
    float x = readAxis(0x32) - offset_x;
    float y = readAxis(0x34) - offset_y;
    float z = readAxis(0x36) - offset_z;

    float x_g = x / 256.0;
    float y_g = y / 256.0;
    float z_g = z / 256.0;

    float raw_roll = atan2(y_g, sqrt(x_g * x_g + z_g * z_g)) * 180.0 / PI;

    //==================дҪҺйҖҡжҝҫжіў==================
    filtered_roll = alpha_roll * raw_roll + (1 - alpha_roll) * filtered_roll;

    //==================йҷҗеҲ¶зҜ„еңҚ==================
    filtered_roll = constrain(filtered_roll, -60, 60);

    //================== жҳ е°„еҲ° Servo еҸҜеӢ•зҜ„еңҚпјҲ30~150еәҰпјү==================
    int servo_angle = (int)((filtered_roll + 60.0) * (150.0 - 30.0) / 120.0 + 30.0);
    servo_angle = constrain(servo_angle, 30, 150);

    // ================== й©…еӢ• Servo ==================
    myServo.write(servo_angle);

    //================== еәҸеҲ—еҹ ијёеҮә ==================
    Serial.print("Raw: ");
    Serial.print(raw_roll, 1);
    Serial.print("  Filtered: ");
    Serial.print(filtered_roll, 1);
    Serial.print("  Servo: ");
    Serial.println(servo_angle);
  }
}

// ================== иј”еҠ©еҮҪејҸ ==================
int16_t readAxis(byte reg) {
  Wire.beginTransmission(ADXL345); 
  Wire.write(reg);  // жҢҮеҗ‘ DATAX0 жҡ«еӯҳеҷЁпјҲX и»ёдҪҺдҪҚе…ғзө„пјү
  Wire.endTransmission(false);// дҝқжҢҒйҖЈз·ҡпјҢжә–еӮҷи®ҖеҸ–
  Wire.requestFrom(ADXL345, 2);
    // и«ӢжұӮ 2 еҖӢдҪҚе…ғзө„пјҲLSB + MSBпјү
    // I2C жҜҸж¬ЎеҸӘиғҪеӮі 8 дҪҚе…ғпјҢжүҖд»Ҙ 16 дҪҚе…ғзҡ„ X и»ёж•ёеҖј
    // иў« ADXL345 жӢҶжҲҗе…©еҖӢжҡ«еӯҳеҷЁеҲҶе…©ж¬ЎеӮійҖҒ
  if (Wire.available() >= 2) {// зўәиӘҚ 2 еҖӢдҪҚе…ғзө„йғҪе·Іж”¶еҲ°
    byte lsb = Wire.read();// з¬¬дёҖж¬Ўи®Җ вҶ’ DATAX0 дҪҺдҪҚе…ғзө„
    byte msb = Wire.read();// з¬¬дәҢж¬Ўи®Җ вҶ’ DATAX1 й«ҳдҪҚе…ғзө„
    // е°Үе…©еҖӢдҪҚе…ғзө„йҮҚж–°жӢјеҗҲжҲҗе®Ңж•ҙзҡ„ 16 дҪҚе…ғж•ёеҖј
    // зӮәд»ҖйәјиҰҒжӢҶй–ӢеҸҲеҗҲиө·дҫҶпјҹ
    // еӣ зӮә I2C зЎ¬й«”йҷҗеҲ¶жҜҸж¬ЎеҸӘиғҪеӮі 8 дҪҚе…ғ
    // ADXL345 иў«иҝ«жҠҠ 16 дҪҚе…ғж•ёеҖјжӢҶжҲҗе…©еҚҠеҲҶе…©ж¬ЎйҖҒеҮә
    // жүҖд»ҘжҲ‘еҖ‘ж”¶еҲ°еҫҢеҝ…й ҲиҮӘе·ұжӢјеӣһдҫҶпјҡ
    //
    return (int16_t)((msb << 8) | lsb);
  }
  return 0;
}

//==================ж ЎжӯЈйҡҺж®ө жүҫoffset==================
void cailbrate() {
  Serial.print("ж ЎжӯЈдёӯ дҝқжҢҒйқңжӯў...");
  long sum_x = 0, sum_y = 0, sum_z = 0;
  for (int i = 0; i < 500; i++) {
    sum_x += readAxis(0x32);
    sum_y += readAxis(0x34);
    sum_z += readAxis(0x36);
    delay(10);
  }
  offset_x = sum_x / 500;
  offset_y = sum_y / 500;
  offset_z = sum_z / 500 - 256;
  Serial.println("ж ЎжӯЈе®ҢжҲҗ");
}
