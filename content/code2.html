// ==========================================
// 接收端：NodeMCU ESP8266
// 功能：透過 nRF24L01 接收 STM32 傳來的
//       Pitch / Yaw 角度，控制兩個伺服馬達
// ==========================================


#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>
#include <Servo.h>

// ================== nRF24L01 ==================
// CE=D4(GPIO2), CSN=D8(GPIO15)
// SCK=D5(GPIO14), MOSI=D7(GPIO13), MISO=D6(GPIO12)
RF24 radio(D4, D8);
const byte address[6] = "00001";  // 必須與發射端完全相同

// ================== Servo ==================
Servo pitchServo;  // D1 (GPIO5)
Servo yawServo;    // D2 (GPIO4)

// ================== 傳輸封包 ==================
// 必須與發射端結構完全一致，共 8 bytes
struct AnglePacket {
  float pitch;  // 俯仰角 -60~+60 度
  float yaw;    // 偏航角 -60~+60 度
};

// ================== 接收端濾波 ==================
// 補償無線偶發跳值，alpha 與發射端獨立
float last_pitch = 0.0;
float last_yaw   = 0.0;
const float alpha = 0.5;  // 0.3~0.6，越大反應越快

// ==========================================
//              Setup
// ==========================================
void setup() {
  Serial.begin(115200);



  // Servo 初始化
  // 500~2400us 脈衝範圍，相容大多數舵機
  pitchServo.attach(D1, 500, 2400);  // GPIO5
  yawServo.attach(D2, 500, 2400);    // GPIO4
  pitchServo.write(90);              // 上電歸中
  yawServo.write(90);
  Serial.println("Servo ready (D1=Pitch, D2=Yaw)");

  // nRF24L01 初始化
radio.begin();
  radio.setPALevel(RF24_PA_HIGH);     // ← 與發送端一致
  radio.setDataRate(RF24_1MBPS);      // ← 改成 1MBPS
  radio.setChannel(108);
  radio.setAutoAck(false);
  radio.setPayloadSize(8);
  radio.openReadingPipe(0, address);
  radio.startListening();

  Serial.println("nRF24L01 RX ready (HIGH + 1MBPS)");
}

// ==========================================
//              Loop
// ==========================================
void loop() {

  // 有封包就立即處理（非阻塞）
  if (radio.available()) {
    AnglePacket pkt;
    radio.read(&pkt, sizeof(pkt));  // 讀取 8 bytes

    // 低通濾波：平滑偶發跳值
    last_pitch = alpha * pkt.pitch + (1.0 - alpha) * last_pitch;
    last_yaw   = alpha * pkt.yaw   + (1.0 - alpha) * last_yaw;

    // 角度映射：-60~+60 度 → Servo 45~135 度
    // 保留機械緩衝，避免撞到極限位置
    // 建議把 constrain 改成 float 版本，避免精度損失
    int sp = map(constrain(last_pitch, -60.0f, 60.0f), -60, 60, 45, 135);
    int sy = map(constrain(last_yaw,   -60.0f, 60.0f), -60, 60, 45, 135);

    pitchServo.write(sp);
    yawServo.write(sy);

    // Serial 節流：每 50ms 輸出一次
    static uint32_t lastPrint = 0;
    if (millis() - lastPrint >= 50) {
      lastPrint = millis();
      Serial.print("Pitch: "); Serial.print(last_pitch, 1);
      Serial.print("  Yaw: "); Serial.print(last_yaw, 1);
      Serial.print("  SP: ");  Serial.print(sp);
      Serial.print("  SY: ");  Serial.println(sy);
    }
  }
}
