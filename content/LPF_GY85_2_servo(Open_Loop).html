LPF_GY85_2_servo(Open_Loop)

matlab每次重新跑偏移值都會變，所以接下來就是把它改成閉迴路。

開迴路系統：GY-85 → 低通濾波 → 轉 Servo 指令角 → Servo 動作 → ADC 回授 → 中位數 + 低通濾波 → 轉實際角度 → MATLAB 分析誤差 → 加補償

ARDUINO CODE
#include <Wire.h>
#include <Servo.h>

Servo myServo;

unsigned long lastTime = 0;

#define ADXL345 0x53

int16_t offset_x = 0, offset_y = 0, offset_z = 0;

// ================== 濾波相關變數 ==================
float filtered_roll = 0.0;        // EMA 濾波後的 roll
float alpha_roll = 0.24;          // 指令端 alpha（可調）

// Servo ADC 濾波
const int median_size = 5;
int adc_buffer[median_size] = {0};//等於0矩陣
int buffer_idx = 0;
float filtered_adc_angle = 90.0;
float alpha_adc = 0.10;           // ADC 端 alpha 為了讓實際角度讀值平滑，去除馬達負載造成的凸波（spike）和雜訊（主要用在 MATLAB 分析）


// ================== 新增：Servo 偏移補償 ==================
float servo_offset = 25.01;     // 這裡填入 MATLAB 算出的平均偏移值

// =================================================
int16_t readAxis(byte reg);
void cailbrate();

void setup() {
  Serial.begin(115200);
  Wire.setSCL(PB8);
  Wire.setSDA(PB9);
  Wire.begin();
  delay(1000);

  // ================== 初始化 ADXL345 ================== 
  Wire.beginTransmission(ADXL345);
  Wire.write(0x2D); Wire.write(0x08); Wire.endTransmission();
  Wire.beginTransmission(ADXL345);
  Wire.write(0x31); Wire.write(0x00); Wire.endTransmission();

  Serial.println("ADXL345 ready");

  myServo.attach(PA0);
  myServo.write(90);
  Serial.println("Servo ready");

  cailbrate();
  
  // ================== 輸出標題 ================== 
  //GY_TO_servo_angle vs ADC_filtered → 可以看指令角度和實際角度的差距（追蹤誤差）
  //filtered_roll vs raw_roll → 可以看低通濾波的效果
  //ADC_raw vs ADC_filtered → 可以看中位數濾波去凸波的效果
  Serial.println("millis, GY_TO_servo_angle, compensated_angle,  ADC_raw,ADC_filtered, filtered_roll,  raw_roll");
  
  delay(500);
  lastTime = micros();
}
void loop() {
   if(micros() - lastTime >= 10000){ // 10000us = 10ms
  lastTime += 10000;

    // ================== 讀取三軸資料 ================== 
    float x = readAxis(0x32) - offset_x;
    float y = readAxis(0x34) - offset_y;
    float z = readAxis(0x36) - offset_z;

    float x_g = x / 256.0;
    float y_g = y / 256.0;
    float z_g = z / 256.0;

    // ================== 計算原始 roll（未濾波） ================== 
    float raw_roll = atan2(y_g, sqrt(x_g * x_g + z_g * z_g)) * 180.0 / PI;

    // ================== 對 GY85 指令端做 低通 濾波 ================== 
    filtered_roll = alpha_roll * raw_roll + (1 - alpha_roll) * filtered_roll;

    // ================== 轉成 servo 角度（建議範圍 30~150，避免撞極限） ================== 
    int GY_TO_servo_angle = map(constrain((int)filtered_roll, -60, 60), -60, 60, 30, 150);

    // ================== 新增補償：加上 servo_offset ==================
    int compensated_angle = GY_TO_servo_angle + (int)servo_offset;   // 四捨五入轉 int
    compensated_angle = constrain(compensated_angle, 30, 150);       // 限制範圍避免撞極限

    // ================== 對 Servo ADC 做 中位數 + 低通 濾波 ================== 
    int raw_adc = analogRead(PA7);
    adc_buffer[buffer_idx] = raw_adc;
    buffer_idx = (buffer_idx + 1) % median_size;//更新索引，讓下一次寫入時指向下一個位置。索引 到達 4 的時候，下一次就會回到 0。

    // ================== 取中位數去凸波 ================== 
    int temp[median_size];
    for (int i = 0; i < median_size; i++) temp[i] = adc_buffer[i];
    sort(temp, median_size);// 排序副程式在最下面
    int median_adc = temp[median_size / 2];// 第 3 個值就是中位數（最中間的）

    // ================== ADC 轉角度 + 低通 平滑 ================== 
    float adc_angle = 0.215583 * median_adc - 18.501444;
    filtered_adc_angle = alpha_adc * adc_angle + (1 - alpha_adc) * filtered_adc_angle;

    // ================== 驅動伺服 ================== 
    myServo.write(compensated_angle);//改成補償後的指令角度

    // ================== 輸出 ==================
    Serial.print(millis());
    Serial.print(",");
    Serial.print(GY_TO_servo_angle);           // 指令角度
    Serial.print(",");
    Serial.print(compensated_angle);           //補償後的指令角度
    Serial.print(",");
    Serial.print(raw_adc);                     // 原始 ADC（觀察凸波）
    Serial.print(",");
    Serial.print((int)filtered_adc_angle);     // 濾波後實際角度
    Serial.print(",");
    Serial.print(filtered_roll, 3);            // 濾波後的 roll
    Serial.print(",");
    Serial.println(raw_roll, 3);               // 原始 roll（未濾波）

  }
}

// ================== 輔助函式 ==================
int16_t readAxis(byte reg) {
  Wire.beginTransmission(ADXL345); 
  Wire.write(reg);  // 指向 DATAX0 暫存器（X 軸低位元組）
  Wire.endTransmission(false);// 保持連線，準備讀取
  Wire.requestFrom(ADXL345, 2);
    // 請求 2 個位元組（LSB + MSB）
    // I2C 每次只能傳 8 位元，所以 16 位元的 X 軸數值
    // 被 ADXL345 拆成兩個暫存器分兩次傳送
  if (Wire.available() >= 2) {// 確認 2 個位元組都已收到
    byte lsb = Wire.read();// 第一次讀 → DATAX0 低位元組
    byte msb = Wire.read();// 第二次讀 → DATAX1 高位元組
    // 將兩個位元組重新拼合成完整的 16 位元數值
    // 為什麼要拆開又合起來？
    // 因為 I2C 硬體限制每次只能傳 8 位元
    // ADXL345 被迫把 16 位元數值拆成兩半分兩次送出
    // 所以我們收到後必須自己拼回來：
    //
    // msb      = 00000010          （高 8 位元）
    // msb << 8 = 00000010 00000000 （推到高位）
    // | lsb    = 00000010 00000001 （填入低 8 位元）
    // 結果     = 513    
    return (int16_t)((msb << 8) | lsb);
  }
  return 0;
}

//校正階段 找offset
void cailbrate() {
  Serial.print("校正中 保持靜止...");
  long sum_x = 0, sum_y = 0, sum_z = 0;
  for (int i = 0; i < 500; i++) {
    sum_x += readAxis(0x32);
    sum_y += readAxis(0x34);
    sum_z += readAxis(0x36);
    delay(10);
  }
  offset_x = sum_x / 500;
  offset_y = sum_y / 500;
  offset_z = sum_z / 500 - 256;
  Serial.println("校正完成");
}

// 小排序函式（用於中位數濾波）
void sort(int arr[], int n) {
  for (int i = 0; i < n - 1; i++) {
    for (int j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        int temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
  }
}



MATLAB CODE

clc; clear; close all;
%% 1. 讀取資料 用 MATLAB 錄製序列埠資料
s = serialport("COM3", 115200);  
configureTerminator(s, "LF");

data = [];
disp("開始錄製，按 Ctrl+C 停止...");

for i = 1:2000  % 錄 2000 筆 = 約 20 秒
    line = readline(s);
    nums = str2double(split(line, ","));
    if length(nums) == 7
        data = [data; nums'];
    end
end

save("servo_data.mat", "data");
disp("錄製完成");

load("servo_data.mat");

t               = data(:,1) / 1000;  % ms 轉成秒
GY_TO_servo_angle  = data(:,2);      % 指令角度
compensated_angle  = data(:,3);      % 補償後指令角度
ADC_raw         = data(:,4);         % 原始 ADC
ADC_filtered    = data(:,5);         % 濾波後實際角度
filtered_roll   = data(:,6);         % 濾波後 roll
raw_roll        = data(:,7);         % 原始 roll

%% 2. 確認濾波效果
figure('Name','濾波效果');
subplot(2,1,1);
plot(t, raw_roll, 'r', t, filtered_roll, 'b', 'LineWidth', 1.5);
legend('raw roll', 'filtered roll');
title('Roll 濾波效果');
xlabel('時間 (s)'); ylabel('角度 (°)');
grid on;

subplot(2,1,2);
plot(t, ADC_raw, 'r', t, ADC_filtered, 'b', 'LineWidth', 1.5);
legend('ADC raw', 'ADC filtered');
title('ADC 濾波效果');
xlabel('時間 (s)'); ylabel('ADC 值');
grid on;

%% 3. 指令 vs 實際角度 + 平均偏移計算（專為補償用）
figure('Name','指令 vs 實際角度 - 偏移補償');

plot(t, GY_TO_servo_angle, 'b', 'LineWidth', 1.5);   
hold on;
plot(t, ADC_filtered, 'r', 'LineWidth', 1.5);

error = ADC_filtered - GY_TO_servo_angle  ;        % 實際 - 指令
mean_offset = mean(error);                       % 補償值
rmse = sqrt(mean(error.^2));
max_error = max(abs(error));

yline(mean_offset, 'k--', 'LineWidth', 1.2);

legend('指令角度 (GY→Servo)', '實際角度 (ADC)', ...
       ['平均偏移 = ' num2str(mean_offset, '%.3f') ' °']);

title('指令 vs 實際角度（建議補償值）');
xlabel('時間 (s)'); ylabel('角度 (°)');
grid on;

% 在 Command Window 顯示結果
fprintf('\n=== Servo 偏移補償建議 ===\n');
fprintf('平均偏移 (Mean Bias)     : %.3f 度\n', mean_offset);
fprintf('均方根誤差 (RMSE)        : %.3f 度\n', rmse);
fprintf('最大絕對誤差             : %.3f 度\n', max_error);

%% 4.指令 vs 實際角度（改用 compensated_angle 來分析補償效果）
figure('Name','補償後 指令 vs 實際角度');

plot(t, compensated_angle, 'b', 'LineWidth', 1.5);   % 使用補償後的角度
hold on;
plot(t, ADC_filtered, 'r', 'LineWidth', 1.5);

error = ADC_filtered - compensated_angle ; 
mean_offset = mean(error);
rmse = sqrt(mean(error.^2));
max_error = max(abs(error));

yline(mean_offset, 'k--', 'LineWidth', 1.2);

legend('補償後指令角度', '實際角度 (ADC)', ...
       ['平均偏移 = ' num2str(mean_offset, '%.3f') ' °']);

title('補償後 指令 vs 實際角度');
xlabel('時間 (s)'); ylabel('角度 (°)');
grid on;

fprintf('\n=== 補償後誤差統計 ===\n');
fprintf('平均偏移     : %.3f 度\n', mean_offset);
fprintf('RMSE         : %.3f 度\n', rmse);
fprintf('最大絕對誤差 : %.3f 度\n', max_error);
