pitch_yaw

#include <Wire.h>
#include <Servo.h>
#include <math.h>

// ================== Servo 定義 ==================
Servo pitchServo;   // PA0
Servo yawServo;     // PA6

unsigned long lastTime = 0;

#define ADXL345_ADDR  0x53
#define HMC5883L_ADDR 0x1E

// ================== 校正偏移 ==================
int16_t accel_offset_x = 0, accel_offset_y = 0, accel_offset_z = 0;
float mag_offset_x = 0, mag_offset_y = 0;

// ================== 零點誤差 ==================
float pitch_zero_offset = 0.0;
float yaw_zero_offset   = 0.0;

// ================== 狀態控制 ==================
bool servo_active = false;

// ================== 濾波變數 ==================
float filtered_pitch = 0.0;
float filtered_roll  = 0.0;
float filtered_yaw   = 0.0;
float alpha_pitch = 0.35;
float alpha_roll  = 0.35;
float alpha_yaw   = 0.35;

// ==========================================
//              函式宣告
// ==========================================
void initADXL345();
void initHMC5883L();
int16_t readAxisLSB(byte addr, byte reg);
void readAccel(float &x_g, float &y_g, float &z_g);
void readMag(float &mx, float &my, float &mz);
void calibrateAccel();
void calibrateMag();
void recordZeroPoint();
int mapAngle(float angle);

// ==========================================
//              Setup
// ==========================================
void setup() {
  Serial.begin(115200);
  Wire.setSCL(PB8);
  Wire.setSDA(PB9);
  Wire.begin();
  delay(1000);

  initADXL345();
  initHMC5883L();

  pitchServo.attach(PA0);
  yawServo.attach(PA6);
  pitchServo.write(90);
  yawServo.write(90);
  Serial.println("Servo ready (PA0=Pitch, PA6=Yaw)");

  calibrateAccel();
  calibrateMag();

  Serial.println("=========================================");
  Serial.println("指令：L = 鎖定Servo / S = 記錄零點並開始");
  Serial.println("=========================================");

  delay(500);
  lastTime = micros();
}

// ==========================================
//              Loop
// ==========================================
void loop() {

  // ================== 鍵盤輸入 ==================
  if (Serial.available()) {
    char cmd = Serial.read();

    // L：鎖定 Servo 在 90 度
    if (cmd == 'L' || cmd == 'l') {
      servo_active = false;
      pitchServo.write(90);
      yawServo.write(90);
      Serial.println(">> Servo 已停在 90 度，請將頭擺正後輸入 'S'");
    }

    // S：記錄零點並開始
    if (cmd == 'S' || cmd == 's') {
      recordZeroPoint();
    }
  }

  // ================== 主迴圈 20ms ==================
  if (micros() - lastTime >= 20000) {
    lastTime += 20000;

    // 讀取加速度計
    float x_g, y_g, z_g;
    readAccel(x_g, y_g, z_g);

    // Pitch & Roll
    float raw_pitch = atan2(-x_g, sqrt(y_g*y_g + z_g*z_g)) * 180.0 / PI;
    float raw_roll  = atan2(y_g, z_g) * 180.0 / PI;

    filtered_pitch = alpha_pitch * raw_pitch + (1.0 - alpha_pitch) * filtered_pitch;
   filtered_roll  = alpha_roll  * raw_roll  + (1.0 - alpha_roll)  * filtered_roll;

    // 讀取磁力計
    float mx, my, mz;
    readMag(mx, my, mz);

    // 傾斜補償（含 Roll）
    float pitch_rad = filtered_pitch * PI / 180.0;
    float roll_rad  = filtered_roll  * PI / 180.0;

    float mx_h = mx * cos(pitch_rad)
               + my * sin(roll_rad) * sin(pitch_rad)
               - mz * cos(roll_rad) * sin(pitch_rad);
    float my_h = my * cos(roll_rad)
               + mz * sin(roll_rad);

    float raw_yaw = atan2(-my_h, mx_h) * 180.0 / PI;
    filtered_yaw  = alpha_yaw * raw_yaw + (1.0 - alpha_yaw) * filtered_yaw;

    // 套用零點誤差
    float corrected_pitch = filtered_pitch - pitch_zero_offset;
    float corrected_yaw   = filtered_yaw   - yaw_zero_offset;

    corrected_pitch = constrain(corrected_pitch, -60, 60);
    corrected_yaw   = constrain(corrected_yaw,   -60, 60);

    // 只有 active 才驅動 Servo
    if (servo_active) {
      pitchServo.write(mapAngle(corrected_pitch));
      yawServo.write(mapAngle(corrected_yaw));
    }

    // 序列埠輸出
    Serial.print("Pitch: "); Serial.print(corrected_pitch, 1);
    Serial.print("  Yaw: "); Serial.print(corrected_yaw, 1);
    Serial.print("  PS: ");  Serial.print(servo_active ? mapAngle(corrected_pitch) : 90);
    Serial.print("  YS: ");  Serial.print(servo_active ? mapAngle(corrected_yaw)   : 90);
    Serial.print("  [");     Serial.print(servo_active ? "RUN" : "LOCK");
    Serial.println("]");
  }
}

// ==========================================
//              初始化
// ==========================================
void initADXL345() {
  Wire.beginTransmission(ADXL345_ADDR);
  Wire.write(0x2D); Wire.write(0x08);
  Wire.endTransmission();

  Wire.beginTransmission(ADXL345_ADDR);
  Wire.write(0x31); Wire.write(0x00);
  Wire.endTransmission();
  Serial.println("ADXL345 ready");
}

void initHMC5883L() {
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x00); Wire.write(0x70); // 8avg, 15Hz
  Wire.endTransmission();

  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x01); Wire.write(0x20); // Gain=1（±1.3 Ga）
  Wire.endTransmission();

  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x02); Wire.write(0x00); // Continuous mode
  Wire.endTransmission();
  Serial.println("HMC5883L ready");
}

// ==========================================
//              讀取
// ==========================================

// ADXL345：LSB first
int16_t readAxisLSB(byte addr, byte reg) {
  Wire.beginTransmission(addr);
  Wire.write(reg);
  Wire.endTransmission(false);
  Wire.requestFrom(addr, (uint8_t)2);
  if (Wire.available() >= 2) {
    byte lsb = Wire.read();
    byte msb = Wire.read();
    return (int16_t)((msb << 8) | lsb);
  }
  return 0;
}

void readAccel(float &x_g, float &y_g, float &z_g) {
  x_g = (readAxisLSB(ADXL345_ADDR, 0x32) - accel_offset_x) / 256.0;
  y_g = (readAxisLSB(ADXL345_ADDR, 0x34) - accel_offset_y) / 256.0;
  z_g = (readAxisLSB(ADXL345_ADDR, 0x36) - accel_offset_z) / 256.0;
}

// HMC5883L：一次讀完 6 bytes，避免資料不同步
// 暫存器順序：X(0x03), Z(0x05), Y(0x07)
void readMag(float &mx, float &my, float &mz) {
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x03);
  Wire.endTransmission(false);
  Wire.requestFrom(HMC5883L_ADDR, (uint8_t)6);

  if (Wire.available() >= 6) {
    int16_t raw_mx = (Wire.read() << 8) | Wire.read();
    int16_t raw_mz = (Wire.read() << 8) | Wire.read(); // Z 在 Y 前面！
    int16_t raw_my = (Wire.read() << 8) | Wire.read();

    mx = raw_mx - mag_offset_x;
    my = raw_my - mag_offset_y;
    mz = raw_mz;//Z 軸通常偏差較小，所以保留原始值
  }
}

// ==========================================
//              校正
// ==========================================
void calibrateAccel() {
  Serial.println("加速度計校正中，保持靜止...");
  long sum_x = 0, sum_y = 0, sum_z = 0;
  for (int i = 0; i < 500; i++) {
    sum_x += readAxisLSB(ADXL345_ADDR, 0x32);
    sum_y += readAxisLSB(ADXL345_ADDR, 0x34);
    sum_z += readAxisLSB(ADXL345_ADDR, 0x36);
    delay(10);
  }
  accel_offset_x = sum_x / 500;
  accel_offset_y = sum_y / 500;
  accel_offset_z = sum_z / 500 - 256;
  Serial.println("加速度計校正完成");
}

void calibrateMag() {
  float mx_max = -99999, mx_min = 99999;
  float my_max = -99999, my_min = 99999;//一開始設極大/極小值，確保第一次更新一定會改變

  // 第一步：等待開始
  Serial.println("磁力計校正：輸入任意鍵開始擺頭 5 次 ...");
  while (!Serial.available());
  Serial.read();

  Serial.println("開始擺頭 5 次！完成後再按任意鍵結束...");

  // 第二步：持續收資料，直到再按一次鍵
  while (true) {
    // 有按鍵就結束
    if (Serial.available()) {
      Serial.read();
      break;
    }

    Wire.beginTransmission(HMC5883L_ADDR);
    Wire.write(0x03);
    Wire.endTransmission(false);
    Wire.requestFrom(HMC5883L_ADDR, (uint8_t)6);

    if (Wire.available() >= 6) {
      float raw_mx = (int16_t)((Wire.read() << 8) | Wire.read());
      (void)((Wire.read() << 8) | Wire.read()); // skip Z
      float raw_my = (int16_t)((Wire.read() << 8) | Wire.read());

      mx_max = max(mx_max, raw_mx); mx_min = min(mx_min, raw_mx);//max(a, b) 是 Arduino 內建函式，會回傳 a 和 b 中較大的那一個。
      my_max = max(my_max, raw_my); my_min = min(my_min, raw_my);//min(a, b) 回傳兩個數中較小的那一個。
    }

    delay(30);
  }

  // 範圍檢查
  float mx_range = mx_max - mx_min;
  float my_range = my_max - my_min;
  Serial.print("mx 範圍: "); Serial.print(mx_range);
  Serial.print("  my 範圍: "); Serial.println(my_range);

  if (mx_range < 50 || my_range < 50) {
    Serial.println("警告：擺頭角度可能不足，建議重新校正！");
  }

  mag_offset_x = (mx_max + mx_min) / 2.0;
  mag_offset_y = (my_max + my_min) / 2.0;

  Serial.print("Mag offset X: "); Serial.print(mag_offset_x);
  Serial.print("  Y: ");          Serial.println(mag_offset_y);
  Serial.println("磁力計校正完成");
}

// ==========================================
//              零點記錄
// ==========================================
void recordZeroPoint() {
  // 讀取當前角度作為零點
  float x_g, y_g, z_g;
  readAccel(x_g, y_g, z_g);

  float mx, my, mz;
  readMag(mx, my, mz);

  pitch_zero_offset = atan2(-x_g, sqrt(y_g*y_g + z_g*z_g)) * 180.0 / PI;

  float pitch_rad = pitch_zero_offset * PI / 180.0;
  float roll_rad  = atan2(y_g, z_g);

  float mx_h = mx * cos(pitch_rad)
             + my * sin(roll_rad) * sin(pitch_rad)
             - mz * cos(roll_rad) * sin(pitch_rad);

  float my_h = my * cos(roll_rad)
             + mz * sin(roll_rad);
             
  yaw_zero_offset = atan2(-my_h, mx_h) * 180.0 / PI;

  // 濾波初始值歸零，避免開始時突然跳動
  filtered_pitch = 0.0;
  filtered_roll  = 0.0;
  filtered_yaw   = 0.0;

  servo_active = true;

  Serial.print(">> 零點記錄完成 — pitch_offset: ");
  Serial.print(pitch_zero_offset, 1);
  Serial.print("  yaw_offset: ");
  Serial.println(yaw_zero_offset, 1);
  Serial.println(">> Servo 開始跟隨頭部動作");
}

// ==========================================
//              工具函式
// ==========================================
int mapAngle(float angle) {
  return constrain((int)(angle + 90.0), 30, 150);
}
