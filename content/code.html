#include <Wire.h>
#include <SPI.h>
#include <RF24.h>
#include <math.h>

// ================== nRF24L01 ==================
// CE=PA1, CSN=PA4, SCK=PA5, MOSI=PA7, MISO=PA6
RF24 radio(PA1, PA4);
const byte address[6] = "00001";

// ================== 感測器位址 ==================
#define ADXL345_ADDR  0x53
#define HMC5883L_ADDR 0x1E

// ================== 按鈕 / LED ==================
#define BTN_PIN  PA2
#define LED_PIN  PC13  // 低電位亮

// ================== 校正偏移 ==================
int16_t accel_offset_x = 0, accel_offset_y = 0, accel_offset_z = 0;
float   mag_offset_x = 0, mag_offset_y = 0;

// ================== 零點 ==================
float pitch_zero_offset = 0.0;
float yaw_zero_offset   = 0.0;

// ================== 狀態 ==================
bool servo_active = false;

// ================== 濾波 ==================
float filtered_pitch = 0.0;
float filtered_roll  = 0.0;
float filtered_yaw   = 0.0;
const float alpha_pitch = 0.35;
const float alpha_roll  = 0.35;
const float alpha_yaw   = 0.35;

// ================== 按鈕 ==================
enum BtnEvent { BTN_NONE, BTN_SINGLE, BTN_DOUBLE, BTN_LONG };
bool     btn_last_state   = HIGH;
uint32_t btn_press_time   = 0;
uint32_t btn_release_time = 0;
uint8_t  btn_click_count  = 0;
bool     btn_waiting      = false;
const uint16_t LONG_PRESS_MS = 1000;
const uint16_t DOUBLE_GAP_MS = 300;
const uint16_t DEBOUNCE_MS   = 20;

// ================== LED 閃爍 ==================
bool     led_blinking    = false;
uint32_t led_last_toggle = 0;
const uint16_t BLINK_MS  = 300;

// ================== 傳輸封包 ==================
struct AnglePacket {
  float pitch;  // 俯仰角 -60~+60 度
  float yaw;    // 偏航角 -60~+60 度
};

unsigned long lastTime = 0;

// ==========================================
//              函式宣告
// ==========================================
void     initADXL345();
void     initHMC5883L();
int16_t  readAxisLSB(byte addr, byte reg);
void     readAccel(float &x_g, float &y_g, float &z_g);
void     readMag(float &mx, float &my, float &mz);
void     calibrateAccel();
void     calibrateMag();
void     recordZeroPoint();
BtnEvent readButton();
void     updateLED();

// ==========================================
//              Setup
// ==========================================
void setup() {
  Serial.begin(115200);
  delay(1000);

  // I2C
  Wire.setSCL(PB8);
  Wire.setSDA(PB9);
  Wire.begin();
  Wire.setClock(400000);          // Fast Mode 400kHz

  // 按鈕 / LED
  pinMode(BTN_PIN, INPUT_PULLUP);
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH);    // PC13 HIGH = 熄滅

  // SPI 腳位指定（STM32 必須明確指定）
  SPI.setMOSI(PA7);
  SPI.setMISO(PA6);
  SPI.setSCLK(PA5);
  SPI.begin();

  // nRF24L01 初始化，失敗時 LED 快閃
  if (!radio.begin()) {
    Serial.println("NRF24L01 初始化失敗！請檢查接線與電源");
    while (1) {
      digitalWrite(LED_PIN, !digitalRead(LED_PIN));
      delay(100);
    }
  }

  radio.setPALevel(RF24_PA_HIGH);  // HIGH 兼顧距離與穩定性
  radio.setDataRate(RF24_1MBPS);   // 1MBPS 延遲低且距離足夠
  radio.setChannel(108);           // 2.508GHz，避開 WiFi 干擾
  radio.setRetries(0, 0);          // 關閉重傳，降低延遲
  radio.setAutoAck(false);         // 單向廣播，不等 ACK
  radio.setPayloadSize(8);         // 固定 8 bytes，省去協議開銷
  radio.openWritingPipe(address);
  radio.stopListening();           // TX 模式
  Serial.println("nRF24L01 TX ready");

  initADXL345();
  initHMC5883L();
  calibrateAccel();
  calibrateMag();

  Serial.println("=========================================");
  Serial.println("單擊 = 磁力計校正");
  Serial.println("雙擊 = 記錄零點並開始傳輸");
  Serial.println("長按 = 鎖定 Servo (傳 0,0)");
  Serial.println("=========================================");

  lastTime = micros();
}

// ==========================================
//              Loop
// ==========================================
void loop() {

  // ── 按鈕事件 ──
  BtnEvent evt = readButton();
  if (evt == BTN_SINGLE) {
    Serial.println(">> [單擊] 磁力計校正");
    calibrateMag();
  } else if (evt == BTN_DOUBLE) {
    recordZeroPoint();
    Serial.println(">> [雙擊] 記錄零點並開始");
  } else if (evt == BTN_LONG) {
    servo_active = false;
    AnglePacket pkt = { 0.0, 0.0 };
    radio.writeFast(&pkt, sizeof(pkt));
    radio.txStandBy();
    Serial.println(">> [長按] Servo 鎖定 90 度");
  }

  // ── LED 更新 ──
  updateLED();

  // ── 主迴圈 5ms（200Hz）──
  if (micros() - lastTime >= 5000) {
    lastTime += 5000;

    // 讀取加速度計
    float x_g, y_g, z_g;
    readAccel(x_g, y_g, z_g);

    // 俯仰、翻滾計算
    float raw_pitch = atan2(-x_g, sqrt(y_g*y_g + z_g*z_g)) * 180.0 / PI;
    float raw_roll  = atan2(y_g, z_g) * 180.0 / PI;

    // 低通濾波
    filtered_pitch = alpha_pitch * raw_pitch + (1.0 - alpha_pitch) * filtered_pitch;
    filtered_roll  = alpha_roll  * raw_roll  + (1.0 - alpha_roll)  * filtered_roll;

    // 讀取磁力計
    float mx, my, mz;
    readMag(mx, my, mz);

    // 傾斜補償
    float pitch_rad = filtered_pitch * PI / 180.0;
    float roll_rad  = filtered_roll  * PI / 180.0;

    float mx_h = mx * cos(pitch_rad)
               + my * sin(roll_rad) * sin(pitch_rad)
               - mz * cos(roll_rad) * sin(pitch_rad);
    float my_h = my * cos(roll_rad)
               + mz * sin(roll_rad);

    // 偏航計算與濾波
    float raw_yaw = atan2(-my_h, mx_h) * 180.0 / PI;
    filtered_yaw  = alpha_yaw * raw_yaw + (1.0 - alpha_yaw) * filtered_yaw;

    // 套用零點，限制範圍
    float corrected_pitch = constrain(filtered_pitch - pitch_zero_offset, -60, 60);
    float corrected_yaw   = constrain(filtered_yaw   - yaw_zero_offset,   -60, 60);

    // 發射封包
    if (servo_active) {
      AnglePacket pkt = { corrected_pitch, corrected_yaw };
      radio.writeFast(&pkt, sizeof(pkt));
      radio.txStandBy();  // 確保發送完成再繼續
    }

    // Serial 節流：每 50ms 輸出一次
    static uint32_t lastPrint = 0;
    if (millis() - lastPrint >= 50) {
      lastPrint = millis();
      Serial.print("Pitch: "); Serial.print(corrected_pitch, 1);
      Serial.print("  Yaw: "); Serial.print(corrected_yaw, 1);
      Serial.print("  [");     Serial.print(servo_active ? "TX" : "LOCK");
      Serial.println("]");
    }
  }
}

// ==========================================
//              感測器初始化
// ==========================================
void initADXL345() {
  // 0x2D = POWER_CTL，0x08 = Measurement Mode
  Wire.beginTransmission(ADXL345_ADDR);
  Wire.write(0x2D); Wire.write(0x08);
  Wire.endTransmission();
  // 0x31 = DATA_FORMAT，0x00 = ±2g 右對齊
  Wire.beginTransmission(ADXL345_ADDR);
  Wire.write(0x31); Wire.write(0x00);
  Wire.endTransmission();
  Serial.println("ADXL345 ready");
}

void initHMC5883L() {
  // 0x00 = Config A，0x70 = 8avg / 15Hz
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x00); Wire.write(0x70);
  Wire.endTransmission();
  // 0x01 = Config B，0x20 = Gain ±1.3Ga
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x01); Wire.write(0x20);
  Wire.endTransmission();
  // 0x02 = Mode，0x00 = Continuous
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x02); Wire.write(0x00);
  Wire.endTransmission();
  Serial.println("HMC5883L ready");
}

// ==========================================
//              感測器讀取
// ==========================================
int16_t readAxisLSB(byte addr, byte reg) {
  Wire.beginTransmission(addr);
  Wire.write(reg);
  Wire.endTransmission(false);  // Repeated Start
  Wire.requestFrom(addr, (uint8_t)2);
  if (Wire.available() >= 2) {
    byte lsb = Wire.read();
    byte msb = Wire.read();
    return (int16_t)((msb << 8) | lsb);
  }
  return 0;
}

void readAccel(float &x_g, float &y_g, float &z_g) {
  // 扣除靜態偏移，除以 256 轉換為 g（±2g / 10-bit 模式）
  x_g = (readAxisLSB(ADXL345_ADDR, 0x32) - accel_offset_x) / 256.0;
  y_g = (readAxisLSB(ADXL345_ADDR, 0x34) - accel_offset_y) / 256.0;
  z_g = (readAxisLSB(ADXL345_ADDR, 0x36) - accel_offset_z) / 256.0;
}

void readMag(float &mx, float &my, float &mz) {
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x03);
  Wire.endTransmission(false);
  Wire.requestFrom(HMC5883L_ADDR, (uint8_t)6);
  if (Wire.available() >= 6) {
    // HMC5883L 暫存器順序：X → Z → Y（注意 Z 在 Y 前）
    int16_t raw_mx = (Wire.read() << 8) | Wire.read();
    int16_t raw_mz = (Wire.read() << 8) | Wire.read();
    int16_t raw_my = (Wire.read() << 8) | Wire.read();
    mx = raw_mx - mag_offset_x;  // 扣除偏移
    my = raw_my - mag_offset_y;
    mz = raw_mz;
  }
}

// ==========================================
//              校正
// ==========================================
void calibrateAccel() {
  Serial.println("加速度計校正中，保持靜止...");
  long sum_x = 0, sum_y = 0, sum_z = 0;
  for (int i = 0; i < 500; i++) {
    sum_x += readAxisLSB(ADXL345_ADDR, 0x32);
    sum_y += readAxisLSB(ADXL345_ADDR, 0x34);
    sum_z += readAxisLSB(ADXL345_ADDR, 0x36);
    delay(10);  // 500次 × 10ms = 約 5 秒
  }
  accel_offset_x = sum_x / 500;
  accel_offset_y = sum_y / 500;
  accel_offset_z = sum_z / 500 - 256;  // Z 軸扣除 1g 重力對應值
  Serial.println("加速度計校正完成");
}

void calibrateMag() {
  float mx_max = -99999, mx_min = 99999;
  float my_max = -99999, my_min = 99999;

  // 第一步：等待單擊，LED 閃爍
  Serial.println("磁力計校正：單擊按鈕開始擺頭 5 次 ...");
  led_blinking = true;
  led_last_toggle = millis();
  digitalWrite(LED_PIN, LOW);   // PC13 LOW = 亮

  while (true) {
    updateLED();
    if (readButton() == BTN_SINGLE) break;
  }

  // 第二步：收資料，LED 長亮
  Serial.println("開始擺頭 5 次！完成後單擊結束...");
  led_blinking = false;
  digitalWrite(LED_PIN, LOW);   // 長亮

  while (true) {
    updateLED();
    if (readButton() == BTN_SINGLE) break;

    Wire.beginTransmission(HMC5883L_ADDR);
    Wire.write(0x03);
    Wire.endTransmission(false);
    Wire.requestFrom(HMC5883L_ADDR, (uint8_t)6);
    if (Wire.available() >= 6) {
      float raw_mx = (int16_t)((Wire.read() << 8) | Wire.read());
      (void)((Wire.read() << 8) | Wire.read()); // 跳過 Z 軸
      float raw_my = (int16_t)((Wire.read() << 8) | Wire.read());
      mx_max = max(mx_max, raw_mx); mx_min = min(mx_min, raw_mx);
      my_max = max(my_max, raw_my); my_min = min(my_min, raw_my);
    }
    delay(30);
  }

  // 完成，LED 熄滅
  led_blinking = false;
  digitalWrite(LED_PIN, HIGH);  // PC13 HIGH = 熄滅

  float mx_range = mx_max - mx_min;
  float my_range = my_max - my_min;
  Serial.print("mx 範圍: "); Serial.print(mx_range);
  Serial.print("  my 範圍: "); Serial.println(my_range);
  if (mx_range < 50 || my_range < 50)
    Serial.println("警告：擺頭角度可能不足，建議重新校正！");

  mag_offset_x = (mx_max + mx_min) / 2.0;
  mag_offset_y = (my_max + my_min) / 2.0;
  Serial.print("Mag offset X: "); Serial.print(mag_offset_x);
  Serial.print("  Y: ");          Serial.println(mag_offset_y);
  Serial.println("磁力計校正完成");
}

// ==========================================
//              零點記錄
// ==========================================
void recordZeroPoint() {
  float x_g, y_g, z_g;
  readAccel(x_g, y_g, z_g);
  float mx, my, mz;
  readMag(mx, my, mz);

  pitch_zero_offset = atan2(-x_g, sqrt(y_g*y_g + z_g*z_g)) * 180.0 / PI;

  float pitch_rad = pitch_zero_offset * PI / 180.0;
  float roll_rad  = atan2(y_g, z_g);

  float mx_h = mx * cos(pitch_rad)
             + my * sin(roll_rad) * sin(pitch_rad)
             - mz * cos(roll_rad) * sin(pitch_rad);
  float my_h = my * cos(roll_rad)
             + mz * sin(roll_rad);

  yaw_zero_offset = atan2(-my_h, mx_h) * 180.0 / PI;

  filtered_pitch = 0.0;
  filtered_roll  = 0.0;
  filtered_yaw   = 0.0;
  servo_active   = true;

  Serial.print(">> 零點完成 pitch_offset: "); Serial.print(pitch_zero_offset, 1);
  Serial.print("  yaw_offset: ");             Serial.println(yaw_zero_offset, 1);
  Serial.println(">> 開始發射角度訊號");
}

// ==========================================
//              工具函式
// ==========================================
BtnEvent readButton() {
  BtnEvent event = BTN_NONE;
  bool state = digitalRead(BTN_PIN);
  uint32_t now = millis();

  // 下降邊緣：按下
  if (btn_last_state == HIGH && state == LOW) {
    delay(DEBOUNCE_MS);
    if (digitalRead(BTN_PIN) == LOW) btn_press_time = now;
  }

  // 上升邊緣：放開
  if (btn_last_state == LOW && state == HIGH) {
    delay(DEBOUNCE_MS);
    if (digitalRead(BTN_PIN) == HIGH) {
      uint32_t held = now - btn_press_time;
      if (held >= LONG_PRESS_MS) {
        event = BTN_LONG;
        btn_click_count = 0;
        btn_waiting = false;
      } else {
        btn_click_count++;
        btn_release_time = now;
        btn_waiting = true;
      }
    }
  }

  // 雙擊視窗超時判斷
  if (btn_waiting && (now - btn_release_time > DOUBLE_GAP_MS)) {
    if (btn_click_count == 1) event = BTN_SINGLE;
    if (btn_click_count >= 2) event = BTN_DOUBLE;
    btn_click_count = 0;
    btn_waiting = false;
  }

  btn_last_state = state;
  return event;
}

void updateLED() {
  if (led_blinking) {
    if (millis() - led_last_toggle >= BLINK_MS) {
      led_last_toggle = millis();
      digitalWrite(LED_PIN, !digitalRead(LED_PIN));
    }
  }
}
