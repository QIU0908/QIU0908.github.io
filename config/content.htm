How to do data analysis

  1. 拿到原始數據。

  2. 先看一眼數據長什麼樣，知道有沒有亂七八糟的東西（例如數字被當文字、很多空值、極端怪值）。

  3. 把數據洗乾淨，明顯錯的極端值，用箱型圖或簡單規則刪/修正（例如電力消耗量負值就刪）。

  4. 決定目標（y）和特徵（X），機器學習就是要用 X 猜 y。

  5. （選做）把數據做一些調整（特徵工程）EX:數字太大或太小 → 標準化（減平均除標準差)。目標 y 如果差很大 → 可以先取 log（之後再轉回來。

  6. 把資料分成「訓練用」和「測試用」，把 70～80% 當訓練，20～30% 當測試，測試資料不能拿來訓練，這樣才能知道模型在「沒看過的資料」上表現如何

  7. 選一個模型 + 訓練它

  8. 用測試資料做預測

  9. 算迴歸誤差指標 MAE MSE RMSE MAPE MPE R²

  10. 看結果 + 決定下一步

    看數字：MAE/RMSE 夠小嗎？R² 有沒有超過 0.7～0.8？

    如果不好 → 回到第 3～7 步改（加特徵、換模型、調參數、清洗更乾淨…）

迴歸模型誤差指標完整教學

regression error metric

1.error metric

  • MAE 平均絕對誤差 :對真實值-預測值(誤差)取絕對值後(黃色直線距離)相加，再平均。

    MAE=1n∑i=1n∣真實值i−預測值i∣\text{MAE} = \frac{1}{n} \sum_{i=1}^{n} \left| \text{真實值}_i - \text{預測值}_i \right|

  • MSE 均方誤差 :對誤差平方(正方形面積)相加後再平均。

    MSE=1n∑i=1n(真實值i−預測值i)2\text{MSE} = \frac{1}{n} \sum_{i=1}^{n} \left( \text{真實值}_i - \text{預測值}_i \right)^2

           

  • RMSE 均方根誤差:對誤差平方(正方形面積)相加後，平均再開根號。

    RMSE=1n∑i=1n(真實值i−預測值i)2

  • MPE平均百分比誤差:(實際值-預測值)/實際值*100%後取平均。

    MPE=1n∑i=1n真實值i−預測值i真實值i×100%\text{MPE} = \frac{1}{n} \sum_{i=1}^{n} \frac{\text{真實值}_i - \text{預測值}_i}{\text{真實值}_i} \times 100\%

  • MAPE 平均絕對百分比誤差 :abs((實際值-預測值)/實際值*100%後取平均

    MAPE=1n∑i=1n∣真實值i−預測值i真實值i∣×100%\text{MAPE} = \frac{1}{n} \sum_{i=1}^{n} \left| \frac{\text{真實值}_i - \text{預測值}_i}{\text{真實值}_i} \right| \times 100\%

2.goodness of fit

  • R²決定係數
符號 全名 公式 意思
SST Total Sum of Squares sum((y - mean(y)).^2) y 的總變異
SSR Regression Sum of Squares sum((y_pred - mean(y)).^2) 模型解釋到的變異
SSE Error Sum of Squares sum((y - y_pred).^2) 模型沒解釋到的變異（殘差）

*mean(y)為原始數據取平均， y為某一點的原始數據, y_pred為某一點預測方程式的值。

SST=SSR+SSE;

R2=1−∑(某一點的原始數據−某一點預測方程式的值)2∑(某一點的原始數據−原始數據取平均)2=1−殘差平方和 (SSR)總平方和 (SST)R^2 = 1 - \frac{\sum \left( \text{某一點的原始數據} - \text{某一點預測方程式的值} \right)^2}{\sum \left( \text{某一點的原始數據} - \text{原始數據取平均} \right)^2} = 1 - \frac{\text{殘差平方和 (SSR)}}{\text{總平方和 (SST)}}

Q:數值多少算好

A:

指標 單位 越小越好？ 對大誤差（outlier）的敏感度 什麼情況下「看起來小」才算好？ 常見應用場景建議
MAE 跟 y 相同單位 是 低（robust） 跟 y 的平均值或範圍比，小於 5–15% 左右常被接受 業務容易理解、對大錯不特別懲罰
MSE y 單位的平方 是 極高 很難直接看，通常只跟其他模型比 訓練時用（可微分）
RMSE 跟 y 相同單位 是 高 常比 MAE 大一點，跟 y 平均值比，小於 10–20% 常被接受 最常用（平衡解釋力與懲罰大錯）
MAPE 百分比 (%) 是 中等（但 y 接近 0 會爆炸） <10% 優秀、10–20% 好、20–50% 可接受、>50% 通常偏差 銷售、需求預測（規模無關）
MPE 百分比 (%) 接近 0 最好 中等（有正負） 接近 0（有系統性偏誤時會遠離 0） 檢查是否有 bias

Q:MAE RMSE 數值會差不多嗎?

A:

情況一：誤差都差不多大 err = [1, 1, 1, 1, 1] MAE = 1.0 RMSE = 1.0 ← 幾乎一樣

情況二：有一個很大的離群誤差 err = [1, 1, 1, 1, 10] MAE = 2.8 RMSE = 4.6 ← 差很多，因為 10² = 100 被放大了

Q:R² 決定係數 — 多少算好？

A:

  • R² > 0.9–0.95：優秀（物理模型強的領域）
  • R² 0.8–0.9：良好（大多數工程應用）
  • R² 0.6–0.8：可接受（現場數據噪音大）
  • R² < 0.6：通常不夠好，除非是極端複雜系統

#include<Servo.h>

Servo myservo;

int pos = 0; 

void setup(){
  Serial.begin(115200);
  myservo.attach(PA0);
  
  delay(1000);

}

void loop() {
  for (pos = 60; pos <= 150; pos +=30) { 
    
    myservo.write(pos);             
    delay(1000);     

    myservo.detach();

    Serial.print(pos);
    Serial.print(",");
    Serial.println(analogRead(PA7));
    delay(100);   
    myservo.attach(PA0);                    
  }
  for (pos = 120; pos >= 30; pos -= 30) {
    myservo.write(pos);  
    delay(1000);  

    myservo.detach();
    Serial.print(pos);
    Serial.print(",");
    Serial.println(analogRead(PA7));
    delay(100);       
    myservo.attach(PA0);               
  }
}

clc; clear; close all;

figPos = [100 100 900 600];   % 視窗大小，可調
offset = 30;                  % 每個視窗偏移，避免完全重疊
%% 讀取校正資料
org_data = readmatrix('servo_angle_ADC.txt', 'Delimiter', ',');
angle_raw = org_data(:,1);
adc_raw   = org_data(:,2);

%% 先按角度平均（去除重複影響）
unique_angles = unique(angle_raw);%unique();把()中的矩陣的每個元素小到大排列
adc_avg = zeros(size(unique_angles));
count    = zeros(size(unique_angles));%用來計算每個角度有幾筆數據
for i = 1:length(unique_angles)
    mask = (angle_raw == unique_angles(i));%這個角度出現的所有位置 回傳true ...
    % 讓mask變成一邏輯矩陣 EX[0 ,1 ,0 ,0]
    adc_avg(i) = mean(adc_raw(mask));%把這個角度出現的raw 給ADC用
    count(i)    = sum( mask);
end

angle = unique_angles';   % 轉成列向量（ polyfit 喜歡列向量）
adc   = adc_avg';
%% 顯示平均結果
fprintf('\n=== 平均結果 ===\n');
fprintf('角度\t筆數\t平均 ADC\n');
for i = 1:length(unique_angles)
    
        fprintf('%d°\t%d\t%.2f\n', unique_angles(i), count(i), adc_avg(i));
    
end
%% 畫長條圖
figure('Name', '原始散佈圖','Position',figPos + [0 5*offset 0 0]);
bar( unique_angles, adc_avg, 'FaceColor', [0.3 0.7 0.9]);
xlabel('角度 (度)');
ylabel('平均 ADC 值');
title('各角度平均 ADC 值');
grid on;
set(gca, 'XTick', unique_angles, 'XTickLabelRotation', 45);

%% 讀取測試資料
data = readmatrix('find_data.txt', 'Delimiter', ',');
ADC_test = data(:,2);
%% 畫原始散佈圖
figure('Name','Angle vs ADC Calibration','Position',figPos + [0 4*offset 0 0]);
scatter(adc, angle, 60, 'filled', 'MarkerFaceColor','b');
xlabel('ADC value'); ylabel('Angle (deg)');
title('Calibration Data: Angle vs ADC');
set(gca, 'FontSize', 12);

%% 擬合模型：angle = f(ADC)
p_lin = polyfit(adc, angle, 1);   % 一次曲線
p_quad = polyfit(adc, angle, 2);  % 二次曲線

%% 產生平滑曲線用
x_fit = linspace(min(adc), max(adc), 300);
y_lin  = polyval(p_lin,  x_fit);
y_quad = polyval(p_quad, x_fit);

hold on;
plot(x_fit, y_lin,  'r-', 'LineWidth', 2.2, 'DisplayName', 'Linear fit');
plot(x_fit, y_quad, 'g-', 'LineWidth', 2.2, 'DisplayName', 'Quadratic fit');
legend('show', 'Location','northwest');
hold off;

%% 顯示擬合方程式（注意 polyfit 最高次在前）
fprintf('=== 擬合結果 ===\n');
fprintf('一次擬合： angle = %.6f × ADC + %.6f\n', p_lin(1), p_lin(2));
fprintf('二次擬合： angle = %.6e × ADC² + %.6f × ADC + %.6f\n', ...
        p_quad(1), p_quad(2), p_quad(3));

%% 在校正點上計算預測角度 & 誤差指標
angle_pred_lin  = polyval(p_lin,  adc);
angle_pred_quad = polyval(p_quad, adc);

err_lin  = angle_pred_lin  - angle;
err_quad = angle_pred_quad - angle;

mae_lin  = mean(abs(err_lin));
mae_quad = mean(abs(err_quad));

rmse_lin  = sqrt(mean(err_lin.^2));
rmse_quad = sqrt(mean(err_quad.^2));

fprintf('\n=== 校正點上的誤差 (單位：度) ===\n');
fprintf('一次擬合 → MAE = %.4f°   RMSE = %.4f°\n', mae_lin, rmse_lin);
fprintf('二次擬合 → MAE = %.4f°   RMSE = %.4f°\n', mae_quad, rmse_quad);

%% R² (決定係數) — 以角度為 y
SS_tot = sum((angle - mean(angle)).^2);
SS_res_lin  = sum(err_lin.^2);
SS_res_quad = sum(err_quad.^2);
R2_lin  = 1 - SS_res_lin  / SS_tot;
R2_quad = 1 - SS_res_quad / SS_tot;

fprintf('\nR² (linear)  = %.6f\n', R2_lin);
fprintf('R² (quadratic) = %.6f\n', R2_quad);

%% 對測試資料反推角度
fprintf('\n=== 測試資料反推角度 ===\n');
angle_test_lin  = polyval(p_lin,  ADC_test);
angle_test_quad = polyval(p_quad, ADC_test);

%% 看所有預測數據
% 全部資料 (linear)
disp('所有預測結果 (linear)：');
disp('   ADC       預測角度 (線性)');
disp([ADC_test angle_test_lin]);

% 全部資料 (quadratic)
disp(' ');
disp('所有預測結果 (quadratic)：');
disp('   ADC       預測角度 (二次)');
disp([ADC_test angle_test_quad]);
