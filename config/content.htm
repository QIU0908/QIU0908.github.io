How to do data analysis

  1. 拿到原始數據。

  2. 先看一眼數據長什麼樣，知道有沒有亂七八糟的東西（例如數字被當文字、很多空值、極端怪值）。

  3. 把數據洗乾淨，明顯錯的極端值，用箱型圖或簡單規則刪/修正（例如電力消耗量負值就刪）。

  4. 決定目標（y）和特徵（X），機器學習就是要用 X 猜 y。

  5. （選做）把數據做一些調整（特徵工程）EX:數字太大或太小 → 標準化（減平均除標準差)。目標 y 如果差很大 → 可以先取 log（之後再轉回來。

  6. 把資料分成「訓練用」和「測試用」，把 70～80% 當訓練，20～30% 當測試，測試資料不能拿來訓練，這樣才能知道模型在「沒看過的資料」上表現如何

  7. 選一個模型 + 訓練它

  8. 用測試資料做預測

  9. 算誤差指標 MAE MSE RMSE MAPE MPE R²

  10. 看結果 + 決定下一步

    看數字：MAE/RMSE 夠小嗎？R² 有沒有超過 0.7～0.8？

    如果不好 → 回到第 3～7 步改（加特徵、換模型、調參數、清洗更乾淨…）

#include<Servo.h>

Servo myservo;

int pos = 0; 

void setup(){
  Serial.begin(115200);
  myservo.attach(PA0);
  
  delay(1000);

}

void loop() {
  for (pos = 60; pos <= 150; pos +=30) { 
    
    myservo.write(pos);             
    delay(1000);     

    myservo.detach();

    Serial.print(pos);
    Serial.print(",");
    Serial.println(analogRead(PA7));
    delay(100);   
    myservo.attach(PA0);                    
  }
  for (pos = 120; pos >= 30; pos -= 30) {
    myservo.write(pos);  
    delay(1000);  

    myservo.detach();
    Serial.print(pos);
    Serial.print(",");
    Serial.println(analogRead(PA7));
    delay(100);       
    myservo.attach(PA0);               
  }
}

data = readmatrix('C:\Users\light\Desktop\servo_angle_ADC.txt', 'Delimiter', ',');
angle = data(:,1);
ADC   = data(:,2);

target_angles = 30:10:150;

% 初始化
mean_ADC = zeros(size(target_angles));
count    = zeros(size(target_angles));

for i = 1:length(target_angles)
    tgt = target_angles(i);
    
    % 直接用邏輯索引取出符合的 ADC
    this_ADC = ADC(angle == tgt);
    
    % 如果有資料
    if ~isempty(this_ADC)
        % 移除可能的 NaN（
        this_ADC = this_ADC(~isnan(this_ADC));
        
        if ~isempty(this_ADC)
            mean_ADC(i) = mean(this_ADC);
            count(i)    = length(this_ADC);
            
            % 顯示原始資料
            fprintf('\n角度 %d° 的資料（共 %d 筆）：\n', tgt, count(i));
            fprintf('angle = %g, ADC = %g\n', [tgt*ones(count(i),1) this_ADC]');
        end
    else
        fprintf('角度 %d° 無資料\n', tgt);
    end
end

% 顯示平均結果
fprintf('\n=== 平均結果 ===\n');
fprintf('角度\t筆數\t平均 ADC\n');
for i = 1:length(target_angles)
    if count(i) > 0
        fprintf('%d°\t%d\t%.2f\n', target_angles(i), count(i), mean_ADC(i));
    else
        fprintf('%d°\t0\t無資料\n', target_angles(i));
    end
end

% 畫長條圖
figure;
bar(target_angles, mean_ADC, 'FaceColor', [0.3 0.7 0.9]);
xlabel('角度 (度)');
ylabel('平均 ADC 值');
title('各角度平均 ADC 值');
grid on;
set(gca, 'XTick', target_angles, 'XTickLabelRotation', 45);


text(target_angles, mean_ADC, sprintfc('%.1f', mean_ADC), ...
     'HorizontalAlignment', 'center', 'VerticalAlignment', 'bottom');
