How to do data analysis

  1. 拿到原始數據。

  2. 先看一眼數據長什麼樣，知道有沒有亂七八糟的東西（例如數字被當文字、很多空值、極端怪值）。

  3. 把數據洗乾淨，明顯錯的極端值，用箱型圖或簡單規則刪/修正（例如電力消耗量負值就刪）。

  4. 決定目標（y）和特徵（X），機器學習就是要用 X 猜 y。

  5. （選做）把數據做一些調整（特徵工程）EX:數字太大或太小 → 標準化（減平均除標準差)。目標 y 如果差很大 → 可以先取 log（之後再轉回來。

  6. 把資料分成「訓練用」和「測試用」，把 70～80% 當訓練，20～30% 當測試，測試資料不能拿來訓練，這樣才能知道模型在「沒看過的資料」上表現如何

  7. 選一個模型 + 訓練它

  8. 用測試資料做預測

  9. 算迴歸誤差指標 MAE MSE RMSE MAPE MPE R²

  10. 看結果 + 決定下一步

    看數字：MAE/RMSE 夠小嗎？R² 有沒有超過 0.7～0.8？

    如果不好 → 回到第 3～7 步改（加特徵、換模型、調參數、清洗更乾淨…）

迴歸模型誤差指標完整教學

regression error metric

1.error metric

  • MAE 平均絕對誤差 :對真實值-預測值(誤差)取絕對值後(黃色直線距離)相加，再平均。

    MAE=1n∑i=1n∣真實值i−預測值i∣\text{MAE} = \frac{1}{n} \sum_{i=1}^{n} \left| \text{真實值}_i - \text{預測值}_i \right|

  • MSE 均方誤差 :對誤差平方(正方形面積)相加後再平均。

    MSE=1n∑i=1n(真實值i−預測值i)2\text{MSE} = \frac{1}{n} \sum_{i=1}^{n} \left( \text{真實值}_i - \text{預測值}_i \right)^2

           

  • RMSE 均方根誤差:對誤差平方(正方形面積)相加後，平均再開根號。

    RMSE=1n∑i=1n(真實值i−預測值i)2

  • MPE平均百分比誤差:(實際值-預測值)/實際值*100%後取平均。

    MPE=1n∑i=1n真實值i−預測值i真實值i×100%\text{MPE} = \frac{1}{n} \sum_{i=1}^{n} \frac{\text{真實值}_i - \text{預測值}_i}{\text{真實值}_i} \times 100\%

  • MAPE 平均絕對百分比誤差 :abs((實際值-預測值)/實際值*100%後取平均

    MAPE=1n∑i=1n∣真實值i−預測值i真實值i∣×100%\text{MAPE} = \frac{1}{n} \sum_{i=1}^{n} \left| \frac{\text{真實值}_i - \text{預測值}_i}{\text{真實值}_i} \right| \times 100\%

2.goodness of fit

  • R²決定係數
符號 全名 公式 意思
SST Total Sum of Squares sum((y - mean(y)).^2) y 的總變異
SSR Regression Sum of Squares sum((y_pred - mean(y)).^2) 模型解釋到的變異
SSE Error Sum of Squares sum((y - y_pred).^2) 模型沒解釋到的變異（殘差）

*mean(y)為原始數據取平均， y為某一點的原始數據, y_pred為某一點預測方程式的值。

SST=SSR+SSE;

R2=1−∑(某一點的原始數據−某一點預測方程式的值)2∑(某一點的原始數據−原始數據取平均)2=1−殘差平方和 (SSR)總平方和 (SST)R^2 = 1 - \frac{\sum \left( \text{某一點的原始數據} - \text{某一點預測方程式的值} \right)^2}{\sum \left( \text{某一點的原始數據} - \text{原始數據取平均} \right)^2} = 1 - \frac{\text{殘差平方和 (SSR)}}{\text{總平方和 (SST)}}

Q:數值多少算好

A:

指標 單位 越小越好？ 對大誤差（outlier）的敏感度 什麼情況下「看起來小」才算好？ 常見應用場景建議
MAE 跟 y 相同單位 是 低（robust） 跟 y 的平均值或範圍比，小於 5–15% 左右常被接受 業務容易理解、對大錯不特別懲罰
MSE y 單位的平方 是 極高 很難直接看，通常只跟其他模型比 訓練時用（可微分）
RMSE 跟 y 相同單位 是 高 常比 MAE 大一點，跟 y 平均值比，小於 10–20% 常被接受 最常用（平衡解釋力與懲罰大錯）
MAPE 百分比 (%) 是 中等（但 y 接近 0 會爆炸） <10% 優秀、10–20% 好、20–50% 可接受、>50% 通常偏差 銷售、需求預測（規模無關）
MPE 百分比 (%) 接近 0 最好 中等（有正負） 接近 0（有系統性偏誤時會遠離 0） 檢查是否有 bias

Q:MAE RMSE 數值會差不多嗎?

A:

情況一：誤差都差不多大 err = [1, 1, 1, 1, 1] MAE = 1.0 RMSE = 1.0 ← 幾乎一樣

情況二：有一個很大的離群誤差 err = [1, 1, 1, 1, 10] MAE = 2.8 RMSE = 4.6 ← 差很多，因為 10² = 100 被放大了

Q:R² 決定係數 — 多少算好？

A:

  • R² > 0.9–0.95：優秀（物理模型強的領域）
  • R² 0.8–0.9：良好（大多數工程應用）
  • R² 0.6–0.8：可接受（現場數據噪音大）
  • R² < 0.6：通常不夠好，除非是極端複雜系統

matlab每次重新跑偏移值都會變，所以接下來就是把它改成閉迴路。

開迴路系統：GY-85 → 低通濾波 → 轉 Servo 指令角 → Servo 動作 → ADC 回授 → 中位數 + 低通濾波 → 轉實際角度 → MATLAB 分析誤差 → 加補償

Servo myServo;

float alpha_roll = 0.24;          // 指令端 alpha（可調）

// Servo ADC 濾波
const int median_size = 5;

float alpha_adc = 0.10;           // ADC 端 alpha 為了讓實際角度讀值平滑，去除馬達負載造成的凸波（spike）和雜訊（主要用在 MATLAB 分析）

  //filtered_roll vs raw_roll → 可以看低通濾波的效果
  //ADC_raw vs ADC_filtered → 可以看中位數濾波去凸波的效果
  Serial.println("millis, GY_TO_servo_angle, compensated_angle,  ADC_raw,ADC_filtered, filtered_roll,  raw_roll");

void loop() {
   if(micros() - lastTime >= 10000){ // 10000us = 10ms
  lastTime += 10000;

    // ================== 讀取三軸資料 ================== 
    float x = readAxis(0x32) - offset_x;
    float y = readAxis(0x34) - offset_y;
    float z = readAxis(0x36) - offset_z;

    float x_g = x / 256.0;
    float y_g = y / 256.0;
    float z_g = z / 256.0;

    // ================== 計算原始 roll（未濾波） ================== 
    float raw_roll = atan2(y_g, sqrt(x_g * x_g + z_g * z_g)) * 180.0 / PI;

    // ================== 對 GY85 指令端做 低通 濾波 ================== 
    filtered_roll = alpha_roll * raw_roll + (1 - alpha_roll) * filtered_roll;

    // ================== 轉成 servo 角度（建議範圍 30~150，避免撞極限） ================== 
    int GY_TO_servo_angle = map(constrain((int)filtered_roll, -60, 60), -60, 60, 30, 150);

    // ================== 新增補償：加上 servo_offset ==================
    int compensated_angle = GY_TO_servo_angle + (int)servo_offset;   // 四捨五入轉 int
    compensated_angle = constrain(compensated_angle, 30, 150);       // 限制範圍避免撞極限

    // ================== 對 Servo ADC 做 中位數 + 低通 濾波 ================== 
    int raw_adc = analogRead(PA7);
    adc_buffer[buffer_idx] = raw_adc;
    buffer_idx = (buffer_idx + 1) % median_size;//更新索引，讓下一次寫入時指向下一個位置。索引 到達 4 的時候，下一次就會回到 0。

    // ================== 取中位數去凸波 ================== 
    int temp[median_size];
    for (int i = 0; i < median_size; i++) temp[i] = adc_buffer[i];
    sort(temp, median_size);// 排序副程式在最下面
    int median_adc = temp[median_size / 2];// 第 3 個值就是中位數（最中間的）

    // ================== ADC 轉角度 + 低通 平滑 ================== 
    float adc_angle = 0.215583 * median_adc - 18.501444;
    filtered_adc_angle = alpha_adc * adc_angle + (1 - alpha_adc) * filtered_adc_angle;

    // ================== 驅動伺服 ================== 
    myServo.write(compensated_angle);//改成補償後的指令角度

    // ================== 輸出 ==================
    Serial.print(millis());
    Serial.print(",");
    Serial.print(GY_TO_servo_angle);           // 指令角度
    Serial.print(",");
    Serial.print(compensated_angle);           //補償後的指令角度
    Serial.print(",");
    Serial.print(raw_adc);                     // 原始 ADC（觀察凸波）
    Serial.print(",");
    Serial.print((int)filtered_adc_angle);     // 濾波後實際角度
    Serial.print(",");
    Serial.print(filtered_roll, 3);            // 濾波後的 roll
    Serial.print(",");
    Serial.println(raw_roll, 3);               // 原始 roll（未濾波）

  }
}

// ================== 輔助函式 ==================
int16_t readAxis(byte reg) {
  Wire.beginTransmission(ADXL345); 
  Wire.write(reg);  // 指向 DATAX0 暫存器（X 軸低位元組）
  Wire.endTransmission(false);// 保持連線，準備讀取
  Wire.requestFrom(ADXL345, 2);
    // 請求 2 個位元組（LSB + MSB）
    // I2C 每次只能傳 8 位元，所以 16 位元的 X 軸數值
    // 被 ADXL345 拆成兩個暫存器分兩次傳送
  if (Wire.available() >= 2) {// 確認 2 個位元組都已收到
    byte lsb = Wire.read();// 第一次讀 → DATAX0 低位元組
    byte msb = Wire.read();// 第二次讀 → DATAX1 高位元組
    // 將兩個位元組重新拼合成完整的 16 位元數值
    // 為什麼要拆開又合起來？
    // 因為 I2C 硬體限制每次只能傳 8 位元
    // ADXL345 被迫把 16 位元數值拆成兩半分兩次送出
    // 所以我們收到後必須自己拼回來：
    //
    // msb      = 00000010          （高 8 位元）
    // msb << 8 = 00000010 00000000 （推到高位）
    // | lsb    = 00000010 00000001 （填入低 8 位元）
    // 結果     = 513    
    return (int16_t)((msb << 8) | lsb);
  }
  return 0;
}

void sort(int arr[], int n) {
  for (int i = 0; i < n - 1; i++) {
    for (int j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        int temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
  }
}

    float raw_roll = atan2(y_g, sqrt(x_g * x_g + z_g * z_g)) * 180.0 / PI;

    // ==================濾波==================
    filtered_roll = alpha_roll * raw_roll + (1 - alpha_roll) * filtered_roll;

    // ==================開迴路指令角度==================
    filtered_roll = constrain(filtered_roll, -60, 60);//filtered_roll範圍±60 

    float GY_TO_servo_angle =
        (filtered_roll + 60.0) * (150.0 - 30.0) / 120.0 + 30.0;//把filtered_roll範圍改成0~120
        //線性映射公式y = (x - xmin) * (ymax - ymin) / (xmax - xmin) + ymin
        //(x - xmin)/ (xmax - xmin)把 x 壓到0~1
        //× (ymax - ymin)放大到新範圍
        //+ ymin +平移
        GY_TO_servo_angle = constrain(GY_TO_servo_angle, 30, 150);

    // ================== ADC 讀取與濾波 ==================
    int raw_adc = analogRead(PA7);
    adc_buffer[buffer_idx] = raw_adc;
    buffer_idx = (buffer_idx + 1) % median_size;

    int temp[median_size];
    for (int i = 0; i < median_size; i++) temp[i] = adc_buffer[i];
    sort(temp, median_size);
    int median_adc = temp[median_size / 2];
    
    float adc_angle = 0.215583 * median_adc - 18.501444 -8.; //加上offset 偏移

    filtered_adc_angle = alpha_adc * adc_angle + (1 - alpha_adc) * filtered_adc_angle;

    // ================== 誤差計算 ==================

    if (abs(GY_TO_servo_angle - prev_target) > 2.0) {
      integral = 0;  // 目標變了，清掉積分
      }
      prev_target = GY_TO_servo_angle;

    float error = GY_TO_servo_angle - filtered_adc_angle;     // 誤差 = 指令角度 - 濾波後實際角度

    if(abs(error) < error_deadband){ //防小誤差一直積分
      error = 0;
      integral *= 0.85;  // 死區內積分衰減
      }
    else{
      error -= (error > 0 ? error_deadband : -error_deadband);
      }

    // ================== PID 計算 ==================
    integral += error * 0.02;                    // 0.02 秒 = 20ms
    float derivative = (error - last_error) / 0.02;

    // 限制積分項（抗積分飽和）
    integral = constrain(integral, -15, 15);

    float pid_output = Kp * error + Ki * integral + Kd * derivative;

   // ================== 最終輸出 ==================
    int compensated_angle = (int)(GY_TO_servo_angle + pid_output);
    compensated_angle = constrain(compensated_angle, 25, 155);                       
    

    // ==================驅動 Servo==================
    myServo.write(compensated_angle);
   

    // ================== 序列埠輸出（方便調參） ==================
    Serial.print(GY_TO_servo_angle, 1);           // 指令角度
    Serial.print(",");
    Serial.print(filtered_adc_angle, 1);          // 實際角度
    Serial.print(",");
    Serial.print(error, 2);                       // 誤差
    Serial.print(",");
    Serial.print(pid_output, 2);                  // PID 輸出值
    Serial.print(",");
    Serial.println(compensated_angle);            // 最終送給 servo 的角度

    last_error = error;
  }
}

// HMC5883L：一次讀完 6 bytes，避免資料不同步
// 暫存器順序：X(0x03), Z(0x05), Y(0x07)
void readMag(float &mx, float &my, float &mz) {
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x03);
  Wire.endTransmission(false);
  Wire.requestFrom(HMC5883L_ADDR, (uint8_t)6);

  if (Wire.available() >= 6) {
    int16_t raw_mx = (Wire.read() << 8) | Wire.read();
    int16_t raw_mz = (Wire.read() << 8) | Wire.read(); // Z 在 Y 前面！
    int16_t raw_my = (Wire.read() << 8) | Wire.read();

    mx = raw_mx - mag_offset_x;
    my = raw_my - mag_offset_y;
    mz = raw_mz;//Z 軸通常偏差較小，所以保留原始值
  }
}

    Wire.endTransmission(false);

      float raw_mx = (int16_t)((Wire.read() << 8) | Wire.read());
      (void)((Wire.read() << 8) | Wire.read()); // skip Z
      float raw_my = (int16_t)((Wire.read() << 8) | Wire.read());

      mx_max = max(mx_max, raw_mx); mx_min = min(mx_min, raw_mx);//max(a, b) 是 Arduino 內建函式，會回傳 a 和 b 中較大的那一個。
      my_max = max(my_max, raw_my); my_min = min(my_min, raw_my);//min(a, b) 回傳兩個數中較小的那一個。
    }

ESP8266 接收端程式

透過 ESP8266 搭配 nRF24L01 無線模組，接收來自 STM32F411 發送端的按鈕訊號，當接收到「開始擺動」訊號時，控制伺服馬達持續左右擺動，接收到「停止擺動」訊號時，伺服馬達停止更新角度。

ESP8266 與 nRF24L01

  • CE → D4 (GPIO2)

  • CSN → D8 (GPIO15)

  • SCK → D5 (GPIO14)

  • MISO → D6 (GPIO12)

  • MOSI → D7 (GPIO13)

  • VCC → 3.3V (必須穩定供電，不能用 5V)

  • GND → GND

ESP8266 與伺服馬達

  • 訊號線 → D1 (GPIO5)

  • VCC → 外部 5V 電源 (建議獨立供電，避免 ESP8266 電壓掉落)

  • GND → 外部電源 GND，並與 ESP8266 GND 共地

#include <SPI.h>          // 引入 SPI 通訊函式庫
#include <nRF24L01.h>     // 引入 nRF24L01 無線模組定義
#include <RF24.h>         // 引入 RF24 函式庫，方便操作 nRF24L01
#include <Servo.h>        // 引入 Servo 函式庫，用來控制伺服馬達

// 建立 RF24 物件，指定 CE 與 CSN 腳位 (ESP8266 的 D4 與 D8)
RF24 radio(D4, D8);

// 建立 Servo 物件
Servo myServo;

// 無線通訊管道地址，必須與發送端一致
const byte address[6] = "00001";

void setup() {
  Serial.begin(115200);       // 初始化序列埠，方便除錯
  myServo.attach(D1);         // 伺服訊號接 D1 (GPIO5)

  // 初始化 nRF24L01 無線模組
  radio.begin();
  radio.openReadingPipe(0, address); // 開啟接收管道
  radio.setPALevel(RF24_PA_LOW);     // 設定功率等級 (低功率)
  radio.setDataRate(RF24_1MBPS);     // 設定傳輸速率 (1Mbps)
  radio.setChannel(76);              // 設定頻道 (必須與發送端一致)
  radio.startListening();            // 啟動接收模式

  Serial.println("ESP8266 接收端啟動 - 等待按鈕訊號...");
}

// 用來存放接收到的按鈕狀態
bool trigger = false;

void loop() {
  // 檢查是否有新的資料可讀取
  if (radio.available()) {
    radio.read(&trigger, sizeof(trigger)); // 讀取 trigger 狀態
    Serial.println(trigger ? "收到：開始擺動" : "收到：停止擺動");
  }

  if (trigger) {
    // 當 trigger 為 true → 伺服持續左右擺動
    static int pos = 0;   // 當前角度
    static int step = 4;  // 每次移動的角度增量

    myServo.write(pos);   // 設定伺服角度
    pos += step;          // 更新角度

    // 當角度到達邊界 (0 或 180) → 改變方向
    if (pos >= 180 || pos <= 0) step = -step;

    delay(15);            // 控制擺動速度
  }
}

#include <Wire.h>
#include <SPI.h>
#include <RF24.h>
#include <math.h>

// ================== nRF24L01 ==================
RF24 radio(PA1, PA4);           // CE=PA1（晶片使能）, CSN=PA4（SPI片選）
const byte address[6] = "00001"; // 收發雙方必須相同的通訊管道位址

// ================== 感測器 I2C 位址 ==================
#define ADXL345_ADDR  0x53      // 加速度計 I2C 位址（SDO接GND時為0x53）
#define HMC5883L_ADDR 0x1E      // 磁力計 I2C 位址（固定0x1E）

// ================== 按鈕 / LED ==================
#define BTN_PIN  PA2            // 常開按鈕腳位（另一端接GND）
#define LED_PIN  PC13           // STM32F411 內建LED，低電位亮

// ================== 加速度計校正偏移 ==================
// 靜止時各軸原始值與理想值的差，在 calibrateAccel() 中計算
int16_t accel_offset_x = 0, accel_offset_y = 0, accel_offset_z = 0;

// ================== 磁力計校正偏移 ==================
// 擺頭校正時計算的硬鐵效應補償值（Hard Iron Offset）
float   mag_offset_x = 0, mag_offset_y = 0;

// ================== 零點偏移 ==================
// 記錄使用者頭部「正視前方」時的角度，作為後續輸出的基準
float pitch_zero_offset = 0.0;  // 俯仰零點
float yaw_zero_offset   = 0.0;  // 偏航零點

// ================== 狀態旗標 ==================
bool servo_active = false;      // true = 正在發射角度資料；false = 鎖定狀態

// ================== 低通濾波變數 ==================
// 使用一階指數移動平均濾波，減少感測器抖動
// filtered = alpha * raw + (1 - alpha) * filtered
// alpha 越小 → 越平滑但反應越慢；alpha 越大 → 越靈敏但抖動越多
float filtered_pitch = 0.0;
float filtered_roll  = 0.0;
float filtered_yaw   = 0.0;
const float alpha_pitch = 0.35; // 俯仰濾波係數
const float alpha_roll  = 0.35; // 翻滾濾波係數
const float alpha_yaw   = 0.35; // 偏航濾波係數

// ================== 按鈕事件定義 ==================
enum BtnEvent { BTN_NONE, BTN_SINGLE, BTN_DOUBLE, BTN_LONG };

// 按鈕內部狀態變數（由 readButton() 使用）
bool     btn_last_state   = HIGH; // 上次按鈕狀態（INPUT_PULLUP 未按為HIGH）
uint32_t btn_press_time   = 0;    // 按下瞬間的時間戳
uint32_t btn_release_time = 0;    // 放開瞬間的時間戳
uint8_t  btn_click_count  = 0;    // 在雙擊判斷視窗內累計的點擊次數
bool     btn_waiting      = false;// 是否正在等待判斷單/雙擊

const uint16_t LONG_PRESS_MS = 1000; // 持續按超過此時間 → 長按
const uint16_t DOUBLE_GAP_MS = 300;  // 兩次點擊間隔在此時間內 → 雙擊
const uint16_t DEBOUNCE_MS   = 20;   // 去彈跳等待時間

// ================== LED 閃爍控制 ==================
bool     led_blinking    = false; // true = 閃爍模式；false = 靜態模式
uint32_t led_last_toggle = 0;     // 上次切換LED狀態的時間戳
const uint16_t BLINK_MS  = 300;   // 閃爍週期（每隔300ms切換一次）

// ================== 無線傳輸封包結構 ==================
// 用 struct 打包兩個 float，共 8 bytes，收發端必須相同
struct AnglePacket {
  float pitch; // 俯仰角（-60 ~ +60 度），corrected_pitch=0 時 Servo 在 90 度
  float yaw;   // 偏航角（-60 ~ +60 度），corrected_yaw=0   時 Servo 在 90 度
};

unsigned long lastTime = 0; // 主迴圈計時基準（micros）

// ==========================================
//              函式宣告
// ==========================================
void initADXL345();
void initHMC5883L();
int16_t readAxisLSB(byte addr, byte reg);
void readAccel(float &x_g, float &y_g, float &z_g);
void readMag(float &mx, float &my, float &mz);
void calibrateAccel();
void calibrateMag();
void recordZeroPoint();
BtnEvent readButton();
void updateLED();

// ==========================================
//              Setup
// ==========================================
void setup() {
  Serial.begin(115200);
  delay(1000); // 等待電源穩定

  // ── I2C 初始化（指定STM32腳位，預設PB6/PB7會衝突）──
  Wire.setSCL(PB8);
  Wire.setSDA(PB9);
  Wire.begin();
  Wire.setClock(400000); // Fast Mode 400kHz，加快感測器讀取速度

  // ── 按鈕 / LED 腳位設定 ──
  pinMode(BTN_PIN, INPUT_PULLUP); // 內建上拉，按下時腳位變LOW
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH);    // PC13 HIGH = 熄滅（低電位亮）

  // ── SPI 腳位指定（STM32必須明確指定，否則使用錯誤預設腳位）──
  SPI.setMOSI(PA7);  // Master Out Slave In
  SPI.setMISO(PA6);  // Master In Slave Out
  SPI.setSCLK(PA5);  // SPI 時鐘
  SPI.begin();

  // ── nRF24L01 初始化，失敗時 LED 快閃提示 ──
  if (!radio.begin()) {
    Serial.println("NRF24L01 初始化失敗！請檢查接線與電源（VCC 3.3V + 大電容）");
    while(1) {
      digitalWrite(LED_PIN, !digitalRead(LED_PIN));
      delay(100); // 快閃 = 硬體異常
    }
  }

  radio.setPALevel(RF24_PA_LOW);  // 發射功率：LOW=測試用，穩定後可改RF24_PA_HIGH
  radio.setDataRate(RF24_1MBPS);  // 資料速率：1Mbps（比250KBPS更快，適合即時控制）
  radio.setChannel(108);          // 頻道108（2.508GHz），避開WiFi常用的1/6/11頻道
  radio.setRetries(15, 15);       // 重傳：最多15次，每次間隔15*250us
  radio.setAutoAck(false);        // 關閉ACK確認（單向廣播，不等待接收端回應）
  radio.openWritingPipe(address); // 設定發射管道位址
  radio.stopListening();          // 切換為發射模式（TX Mode）
  Serial.println("nRF24L01 TX ready");

  // ── 感測器初始化 ──
  initADXL345();
  initHMC5883L();

  // ── 上電自動校正 ──
  calibrateAccel(); // 加速度計靜態校正（約5秒，需保持靜止）
  calibrateMag();   // 磁力計動態校正（需擺頭，由按鈕控制流程）

  Serial.println("=========================================");
  Serial.println("單擊 = 磁力計校正");
  Serial.println("雙擊 = 記錄零點並開始傳輸");
  Serial.println("長按 = 鎖定 Servo (傳 0,0)");
  Serial.println("=========================================");

  lastTime = micros();
}

// ==========================================
//              Loop
// ==========================================
void loop() {

  // ── 按鈕事件處理 ──
  BtnEvent evt = readButton();

  if (evt == BTN_SINGLE) {
    // 單擊：重新校正磁力計（例如換了環境磁場）
    Serial.println(">> [單擊] 磁力計校正");
    calibrateMag();

  } else if (evt == BTN_DOUBLE) {
    // 雙擊：將當前頭部方向設為零點，並開始傳輸
    recordZeroPoint();
    Serial.println(">> [雙擊] 記錄零點並開始");

  } else if (evt == BTN_LONG) {
    // 長按：停止傳輸，送出 0,0 讓接收端 Servo 歸中到 90 度
    servo_active = false;
    AnglePacket pkt = { 0.0, 0.0 };
    radio.write(&pkt, sizeof(pkt));
    Serial.println(">> [長按] Servo 鎖定 90 度");
  }

  // ── LED 閃爍更新（非阻塞，在 updateLED 裡用 millis 計時）──
  updateLED();

  // ── 主感測迴圈：每 10ms 執行一次（100Hz）──
  if (micros() - lastTime >= 10000) {
    lastTime += 10000; // 累加而非重設，避免執行時間抖動造成週期誤差

    // 讀取加速度計原始值（已扣除靜態偏移，單位：g）
    float x_g, y_g, z_g;
    readAccel(x_g, y_g, z_g);

    // 由加速度計計算俯仰角與翻滾角（單位：度）
    // atan2 回傳 -180~+180 度，這裡限縮在有意義的範圍
    float raw_pitch = atan2(-x_g, sqrt(y_g*y_g + z_g*z_g)) * 180.0 / PI;
    float raw_roll  = atan2(y_g, z_g) * 180.0 / PI;

    // 低通濾波：平滑俯仰與翻滾
    filtered_pitch = alpha_pitch * raw_pitch + (1.0 - alpha_pitch) * filtered_pitch;
    filtered_roll  = alpha_roll  * raw_roll  + (1.0 - alpha_roll)  * filtered_roll;

    // 讀取磁力計（已扣除硬鐵偏移）
    float mx, my, mz;
    readMag(mx, my, mz);

    // 傾斜補償：將磁力計讀值投影到水平面
    // 若不補償，頭部俯仰時偏航角會跑掉
    float pitch_rad = filtered_pitch * PI / 180.0;
    float roll_rad  = filtered_roll  * PI / 180.0;

    float mx_h = mx * cos(pitch_rad)
               + my * sin(roll_rad) * sin(pitch_rad)
               - mz * cos(roll_rad) * sin(pitch_rad);
    float my_h = my * cos(roll_rad)
               + mz * sin(roll_rad);

    // 由補償後的磁力計分量計算偏航角
    float raw_yaw = atan2(-my_h, mx_h) * 180.0 / PI;

    // 低通濾波：平滑偏航
    filtered_yaw  = alpha_yaw * raw_yaw + (1.0 - alpha_yaw) * filtered_yaw;

    // 減去零點偏移，並限制在 ±60 度（對應 Servo 30~150 度）
    float corrected_pitch = constrain(filtered_pitch - pitch_zero_offset, -60, 60);
    float corrected_yaw   = constrain(filtered_yaw   - yaw_zero_offset,   -60, 60);

    // 只有 active 狀態才透過 nRF24L01 發射封包
    if (servo_active) {
      AnglePacket pkt = { corrected_pitch, corrected_yaw };
      radio.write(&pkt, sizeof(pkt)); // 非阻塞發射（autoAck關閉，不等回應）
    }

    // 序列埠監控輸出
    Serial.print("Pitch: "); Serial.print(corrected_pitch, 1);
    Serial.print("  Yaw: "); Serial.print(corrected_yaw, 1);
    Serial.print("  [");     Serial.print(servo_active ? "TX" : "LOCK");
    Serial.println("]");
  }
}

void initADXL345() {
  // 暫存器 0x2D（POWER_CTL）：bit3=1 → 進入 Measurement Mode
  Wire.beginTransmission(ADXL345_ADDR);
  Wire.write(0x2D); Wire.write(0x08);
  Wire.endTransmission();

  // 暫存器 0x31（DATA_FORMAT）：0x00 → ±2g，10-bit，右對齊
  Wire.beginTransmission(ADXL345_ADDR);
  Wire.write(0x31); Wire.write(0x00);
  Wire.endTransmission();

  Serial.println("ADXL345 ready");
}

void initHMC5883L() {
  // 暫存器 0x00（Config A）：0x70 → 8次平均、15Hz 輸出速率
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x00); Wire.write(0x70);
  Wire.endTransmission();

  // 暫存器 0x01（Config B）：0x20 → Gain=1，量程 ±1.3 Ga
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x01); Wire.write(0x20);
  Wire.endTransmission();

  // 暫存器 0x02（Mode）：0x00 → Continuous Measurement Mode（持續量測）
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x02); Wire.write(0x00);
  Wire.endTransmission();

  Serial.println("HMC5883L ready");
}

//              感測器讀取

// 從 I2C 裝置讀取 2 bytes，ADXL345 是 LSB 在前
int16_t readAxisLSB(byte addr, byte reg) {
  Wire.beginTransmission(addr);
  Wire.write(reg);
  Wire.endTransmission(false); // Repeated Start，保持 I2C 匯流排控制權
  Wire.requestFrom(addr, (uint8_t)2);
  if (Wire.available() >= 2) {
    byte lsb = Wire.read();
    byte msb = Wire.read();
    return (int16_t)((msb << 8) | lsb); // 合併為 16-bit 有號整數
  }
  return 0;
}

void readAccel(float &x_g, float &y_g, float &z_g) {
  // 讀取三軸原始值，扣除靜態偏移後，除以 256 轉換為 g 單位
  // ADXL345 在 ±2g / 10-bit 模式下，1g ≈ 256 LSB
  x_g = (readAxisLSB(ADXL345_ADDR, 0x32) - accel_offset_x) / 256.0;
  y_g = (readAxisLSB(ADXL345_ADDR, 0x34) - accel_offset_y) / 256.0;
  z_g = (readAxisLSB(ADXL345_ADDR, 0x36) - accel_offset_z) / 256.0;
}

void readMag(float &mx, float &my, float &mz) {
  Wire.beginTransmission(HMC5883L_ADDR);
  Wire.write(0x03); // 從資料暫存器起始位址讀取
  Wire.endTransmission(false);
  Wire.requestFrom(HMC5883L_ADDR, (uint8_t)6); // 一次讀 6 bytes（X/Z/Y 各 2 bytes）

  if (Wire.available() >= 6) {
    // HMC5883L 暫存器順序：X(0x03-04), Z(0x05-06), Y(0x07-08)
    // 注意！Z 在 Y 之前，與一般直覺不同
    int16_t raw_mx = (Wire.read() << 8) | Wire.read(); // X 軸
    int16_t raw_mz = (Wire.read() << 8) | Wire.read(); // Z 軸（順序在Y前）
    int16_t raw_my = (Wire.read() << 8) | Wire.read(); // Y 軸

    mx = raw_mx - mag_offset_x; // 扣除硬鐵偏移
    my = raw_my - mag_offset_y;
    mz = raw_mz;                 // Z 軸偏差通常較小，保留原始值
  }
}

void calibrateAccel() {
  // 靜止採樣 500 次取平均，計算各軸靜態偏移
  // Z 軸需額外減去 256（對應 1g 重力，讓 Z 軸在水平時接近 0）
  Serial.println("加速度計校正中，保持靜止...");
  long sum_x = 0, sum_y = 0, sum_z = 0;
  for (int i = 0; i < 500; i++) {
    sum_x += readAxisLSB(ADXL345_ADDR, 0x32);
    sum_y += readAxisLSB(ADXL345_ADDR, 0x34);
    sum_z += readAxisLSB(ADXL345_ADDR, 0x36);
    delay(10); // 每次間隔 10ms，總計約 5 秒
  }
  accel_offset_x = sum_x / 500;
  accel_offset_y = sum_y / 500;
  accel_offset_z = sum_z / 500 - 256; // 扣掉重力 1g 對應的 256 LSB
  Serial.println("加速度計校正完成");
}

void calibrateMag() {
  // 最大最小值法（Hard Iron Calibration）
  // 讓使用者擺頭，收集各方向磁場值，偏移 = (max + min) / 2
  float mx_max = -99999, mx_min = 99999;
  float my_max = -99999, my_min = 99999;

  // ── 第一步：等待單擊，LED 閃爍提示「請按鈕開始」──
  Serial.println("磁力計校正：單擊按鈕開始擺頭 5 次 ...");
  led_blinking = true;
  led_last_toggle = millis();
  digitalWrite(LED_PIN, LOW); // PC13 LOW = 亮，啟動閃爍

  while (true) {
    updateLED();
    if (readButton() == BTN_SINGLE) break; // 等到單擊才繼續
  }

  // ── 第二步：持續收集磁力計資料，LED 長亮提示「正在收資料」──
  Serial.println("開始擺頭 5 次！完成後單擊結束...");
  led_blinking = false;
  digitalWrite(LED_PIN, LOW); // 停止閃爍，長亮

  while (true) {
    updateLED();
    if (readButton() == BTN_SINGLE) break; // 使用者完成擺頭後單擊結束

    // 直接讀原始磁力計資料（不透過 readMag，避免混用校正值）
    Wire.beginTransmission(HMC5883L_ADDR);
    Wire.write(0x03);
    Wire.endTransmission(false);
    Wire.requestFrom(HMC5883L_ADDR, (uint8_t)6);
    if (Wire.available() >= 6) {
      float raw_mx = (int16_t)((Wire.read() << 8) | Wire.read());
      (void)((Wire.read() << 8) | Wire.read()); // Z 軸跳過
      float raw_my = (int16_t)((Wire.read() << 8) | Wire.read());

      // 即時更新各軸最大最小值
      mx_max = max(mx_max, raw_mx); mx_min = min(mx_min, raw_mx);
      my_max = max(my_max, raw_my); my_min = min(my_min, raw_my);
    }
    delay(30); // 約 33Hz 取樣
  }

  // ── 校正完成，LED 熄滅 ──
  led_blinking = false;
  digitalWrite(LED_PIN, HIGH); // PC13 HIGH = 熄滅

  // 範圍檢查：若擺頭角度不足，範圍會太小，校正精度差
  float mx_range = mx_max - mx_min;
  float my_range = my_max - my_min;
  Serial.print("mx 範圍: "); Serial.print(mx_range);
  Serial.print("  my 範圍: "); Serial.println(my_range);
  if (mx_range < 50 || my_range < 50)
    Serial.println("警告：擺頭角度可能不足，建議重新校正！");

  // 計算並儲存偏移量
  mag_offset_x = (mx_max + mx_min) / 2.0;
  mag_offset_y = (my_max + my_min) / 2.0;
  Serial.print("Mag offset X: "); Serial.print(mag_offset_x);
  Serial.print("  Y: ");          Serial.println(mag_offset_y);
  Serial.println("磁力計校正完成");
}

//              零點記錄
// ==========================================

void recordZeroPoint() {
  // 讀取當前一瞬間的角度作為零點基準
  // 之後所有輸出角度都會減去這個值，使「正視前方」= 0 度
  float x_g, y_g, z_g;
  readAccel(x_g, y_g, z_g);
  float mx, my, mz;
  readMag(mx, my, mz);

  // 計算當前俯仰角作為 pitch 零點
  pitch_zero_offset = atan2(-x_g, sqrt(y_g*y_g + z_g*z_g)) * 180.0 / PI;

  // 用當前俯仰、翻滾對磁力計做傾斜補償，再計算 yaw 零點
  float pitch_rad = pitch_zero_offset * PI / 180.0;
  float roll_rad  = atan2(y_g, z_g);

  float mx_h = mx * cos(pitch_rad)
             + my * sin(roll_rad) * sin(pitch_rad)
             - mz * cos(roll_rad) * sin(pitch_rad);
  float my_h = my * cos(roll_rad)
             + mz * sin(roll_rad);

  yaw_zero_offset = atan2(-my_h, mx_h) * 180.0 / PI;

  // 重置濾波器，避免開始時從舊值突然跳到零點附近
  filtered_pitch = 0.0;
  filtered_roll  = 0.0;
  filtered_yaw   = 0.0;
  servo_active   = true; // 啟動發射

  Serial.print(">> 零點完成 pitch_offset: "); Serial.print(pitch_zero_offset, 1);
  Serial.print("  yaw_offset: ");             Serial.println(yaw_zero_offset, 1);
  Serial.println(">> 開始發射角度訊號");
}

// 按鈕狀態機：偵測單擊、雙擊、長按
// 使用邊緣觸發（狀態改變時才處理），配合計時判斷類型
BtnEvent readButton() {
  BtnEvent event = BTN_NONE;
  bool state = digitalRead(BTN_PIN);
  uint32_t now = millis();

  // 偵測下降邊緣（HIGH→LOW = 按下）
  if (btn_last_state == HIGH && state == LOW) {
    delay(DEBOUNCE_MS);                          // 等待彈跳穩定
    if (digitalRead(BTN_PIN) == LOW)
      btn_press_time = now;                      // 記錄按下時間
  }

  // 偵測上升邊緣（LOW→HIGH = 放開）
  if (btn_last_state == LOW && state == HIGH) {
    delay(DEBOUNCE_MS);                          // 等待彈跳穩定
    if (digitalRead(BTN_PIN) == HIGH) {
      uint32_t held = now - btn_press_time;      // 計算按住時間

      if (held >= LONG_PRESS_MS) {
        // 按住超過門檻 → 長按，立即回傳，不進入單/雙擊流程
        event = BTN_LONG;
        btn_click_count = 0;
        btn_waiting = false;
      } else {
        // 短按放開 → 累加點擊次數，進入等待雙擊視窗
        btn_click_count++;
        btn_release_time = now;
        btn_waiting = true;
      }
    }
  }

  // 雙擊等待視窗超時 → 判斷最終事件
  if (btn_waiting && (now - btn_release_time > DOUBLE_GAP_MS)) {
    if (btn_click_count == 1) event = BTN_SINGLE; // 視窗內只點一下 → 單擊
    if (btn_click_count >= 2) event = BTN_DOUBLE; // 視窗內點兩下以上 → 雙擊
    btn_click_count = 0;
    btn_waiting = false;
  }

  btn_last_state = state;
  return event;
}

// LED 非阻塞閃爍控制（在 loop 中持續呼叫）
// led_blinking = true  → 每 BLINK_MS 切換一次
// led_blinking = false → 不動作（狀態由外部直接 digitalWrite 控制）
void updateLED() {
  if (led_blinking) {
    if (millis() - led_last_toggle >= BLINK_MS) {
      led_last_toggle = millis();
      digitalWrite(LED_PIN, !digitalRead(LED_PIN)); // 反轉，PC13 自動處理極性
    }
  }
}

// ==========================================
// 接收端：NodeMCU ESP8266
// 功能：透過 nRF24L01 接收 STM32 傳來的
//       Pitch / Yaw 角度，控制兩個伺服馬達
// ==========================================

#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>
#include <Servo.h>

// ================== nRF24L01 ==================
// CE=D4(GPIO2), CSN=D8(GPIO15)
// SCK=D5(GPIO14), MOSI=D7(GPIO13), MISO=D6(GPIO12)（SPI預設腳位，不需額外設定）
RF24 radio(D4, D8);

// 通訊管道位址，必須與發射端完全相同
const byte address[6] = "00001";

// ================== Servo ==================
Servo pitchServo;  // 俯仰舵機，接 D1 (GPIO5)
Servo yawServo;    // 偏航舵機，接 D2 (GPIO4)

// ================== 傳輸封包結構 ==================
// 必須與發射端（STM32）定義完全一致，共 8 bytes
struct AnglePacket {
  float pitch; // 俯仰角（-60 ~ +60 度）
  float yaw;   // 偏航角（-60 ~ +60 度）
};

// ================== 接收端濾波變數 ==================
// 對收到的角度再做一次低通濾波，補償無線傳輸偶發的跳值
float last_pitch = 0.0;
float last_yaw   = 0.0;
const float alpha = 0.45; // 平滑係數：越大反應越快，越小越穩；建議範圍 0.3~0.6

// ==========================================
//              Setup
// ==========================================
void setup() {
  Serial.begin(115200);

  // ── Servo 初始化 ──
  // 明確設定脈衝寬度範圍（500~2400us），相容大多數舵機
  // ESP8266 預設 544~2400，部分舵機在 544us 會抖動，改 500 更穩
  pitchServo.attach(D1, 500, 2400); // D1 = GPIO5
  yawServo.attach(D2, 500, 2400);   // D2 = GPIO4
  pitchServo.write(90);             // 上電先歸中，避免突然轉動
  yawServo.write(90);
  Serial.println("Servo ready (D1=Pitch, D2=Yaw)");

  // ── nRF24L01 初始化 ──
  radio.begin();
  radio.setPALevel(RF24_PA_LOW);  // 發射功率，與發射端一致即可（接收端設定不影響收訊）
  radio.setDataRate(RF24_1MBPS);  // 資料速率，必須與發射端相同
  radio.setChannel(108);          // 頻道，必須與發射端相同（2.508GHz）
  radio.setAutoAck(false);        // 關閉 ACK，與發射端一致（單向廣播模式）
  radio.openReadingPipe(0, address); // 開啟接收管道 0，監聽指定位址
  radio.startListening();            // 切換為接收模式（RX Mode）
  Serial.println("nRF24L01 RX ready");
}

// ==========================================
//              Loop
// ==========================================
void loop() {

  // 檢查是否有新封包抵達（非阻塞，無資料直接跳過）
  if (radio.available()) {
    AnglePacket pkt;
    radio.read(&pkt, sizeof(pkt)); // 讀取 8 bytes 封包到 pkt

    // ── 接收端二次低通濾波 ──
    // 平滑封包中偶發的跳值（例如無線干擾造成的突波）
    // filtered = alpha * new_value + (1 - alpha) * old_value
    last_pitch = alpha * pkt.pitch + (1.0 - alpha) * last_pitch;
    last_yaw   = alpha * pkt.yaw   + (1.0 - alpha) * last_yaw;

    // ── 角度轉換為 Servo 角度 ──
    // 輸入：-60 ~ +60 度（頭部偏移量）
    // 輸出：45 ~ 135 度（Servo 行程，中心點 = 90 度）
    // 使用 45~135 而非 0~180，保留機械緩衝，避免舵機撞到極限
    int sp = map(constrain((int)last_pitch, -60, 60), -60, 60, 45, 135);
    int sy = map(constrain((int)last_yaw,   -60, 60), -60, 60, 45, 135);

    pitchServo.write(sp);
    yawServo.write(sy);

    // 序列埠監控輸出
    Serial.print("Pitch: "); Serial.print(last_pitch, 1);
    Serial.print("  Yaw: "); Serial.print(last_yaw, 1);
    Serial.print("  SP: ");  Serial.print(sp);
    Serial.print("  SY: ");  Serial.println(sy);
  }
}
